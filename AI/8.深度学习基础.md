```python
import torch
import matplotlib.pyplot as plt
from sklearn.datasets import make_regression
from torch import nn
from torch.utils.data import TensorDataset, DataLoader

x,y,coef = make_regression(n_samples=1000,n_features=1,bias=5,noise=20,coef=True)
x = torch.tensor(x,dtype=torch.float32)
# 一维张量转为二维，否则计算损失时与y_pred形状不匹配会出错
y = torch.tensor(y,dtype=torch.float32).reshape(-1,1)


dataset = TensorDataset(x,y)
dataloader = DataLoader(dataset=dataset,batch_size=10, shuffle=True)

model = nn.Linear(in_features=1,out_features=1)

criterion = nn.MSELoss()

optimizer = torch.optim.SGD(model.parameters(), lr=0.01)

maxepochs = 10
avg_loss = []
for epoch in range(maxepochs):
    all_loss = 0.0
    sample = 0
    for x_,y_ in dataloader:
        y_pred = model(x_)
        loss = criterion(y_pred,y_)
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
        all_loss += loss.item()
        sample += 1
    avg_loss.append(all_loss/sample)

plt.plot(avg_loss)
plt.show()

x_range = torch.linspace(x.min(), x.max(),100)
y1 = torch.tensor([v*model.weight.item()+model.bias.item() for v in x_range])
y2 = torch.tensor([v*coef+5 for v in x_range])
plt.scatter(x,y,label="data")
plt.plot(x_range,y1,'r-',label='train')
plt.plot(x_range,y2,'b-',label='real')
plt.legend()
plt.show()
```

## 01.激活函数

- **sigmoid**导数数值范围是 **(0, 0.25)**，当**输入 <-6 或者 >6 时，sigmoid 激活函数图像的导数接近为 0**，此时网络参数将更新极其缓慢，或者无法更新。
- **Tanh** 函数将输入**映射到 (-1, 1)** 之间，当**输入大概 < -3 或者 >3 时将被映射为 -1 或者 1**。其**导数值范围 (0, 1)**，当输入的值**大概 < -3 或者 > 3** 时，其导数近似 0。
- **ReLU** 激活函数将小于 0 的值映射为 0，而大于 0 的值则保持不变，运算更为简单，训练效率高。ReLU 能够**在 x>0 时保持梯度不衰减，从而缓解梯度消失问题**。
- **Softmax** 就是将网络输出的 logits 通过 softmax 函数，就映射成为 (0,1) 的值，而这些值的累和为 1，那么我们将它理解成概率，选取概率最大节点，作为我们的预测目标类别。
- 对于输出层：二分类问题选择 **sigmoid** 激活函数，多分类问题选择 **softmax** 激活函数


## 03.model

![1753701674973](C:\Users\17630\AppData\Roaming\Typora\typora-user-images\1753701674973.png)

在 pytorch 中定义深度神经网络其实就是层堆叠的过程，继承自 nn.Module，实现两个方法：

- `__init__`方法：定义网络中的层结构，主要是全连接层，并进行初始化
- forward 方法：定义数据的前向传输，在实例化模型的时候，**底层会自动调用该函数**

```python
import torch
import torch.nn as nn
from torchsummary import summary # 参数统计

# 定义神经网络模型类
class model(nn.Module):
    # 初始化网络层
    def __init__(self):
        super().__init__()
        # 第一层：输入3维，输出3维
        self.layer1 = nn.Linear(in_features=3, out_features=3)
        # 初始化第一层偏置（Kaiming正态分布）
        nn.init.kaiming_normal_(self.layer1.bias)
        # 第二层：输入3维，输出2维
        self.layer2 = nn.Linear(in_features=3, out_features=2)
        # 初始化第二层权重（Xavier均匀分布）
        nn.init.xavier_uniform_(self.layer2.weight)
        # 输出层：输入2维，输出2维
        self.out = nn.Linear(in_features=2, out_features=2)
        # 初始化输出层权重（均匀分布）
        nn.init.uniform_(self.out.weight)

    # 前向传播过程
    def forward(self, x):
        x_layer1 = self.layer1(x)       # 第一层线性变换，x_layer1 = 权重 * x + 偏置
        x_layer1 = torch.sigmoid(x_layer1)  # sigmoid激活函数
        x_layer2 = self.layer2(x_layer1)    # 第二层线性变换
        x_layer2 = torch.relu(x_layer2)     # ReLU激活函数
        out = self.out(x_layer2)        # 输出层线性变换，得到最终的原始预测分数（logits）
        out = torch.softmax(out, dim=-1)    # softmax转换为概率分布
        return out

if __name__ == '__main__':
    my_model = model()  # 实例化模型
    # 生成随机输入（10个样本，每个3维特征）
    x = torch.randn(10, 3)  # [B,V]----->[Batch,vector]
    out = my_model(x)   # 内部会自动触发 forward 方法
    print("输出形状：", out.shape)  # 应输出(10, 2)

    # 打印模型结构摘要（输入大小3，批次大小8）
    summary(my_model, input_size=(3,), batch_size=8)#input_size=(3,):指定单个样本的输入形状

    # 打印模型所有参数名称和值
    for name, para in my_model.named_parameters():
        print("参数名称：", name)
        print("参数值：", para)
        
```



## 04.损失函数

- **Epoch**：使用全部数据对模型进行一次完整训练，训练轮次
- **Batch size**：使用训练集中的小部分样本对模型权重进行一次反向传播的参数更新，每次训练每批次样本数量
- **Iteration**：使用一个 Batch 数据对模型进行一次参数更新的过程（迭代）

假设数据集有 50000 个训练样本，现在选择 Batch Size = 256 对模型进行训练。
每个 Epoch 要训练的图片数量：50000
训练集具有的 Batch 个数：50000/256 + 1 = 196
每个 Epoch 具有的 Iteration 个数：196，10 个 Epoch 具有的 Iteration 个数：1960

```python
import torch.nn as nn
import torch

# 交叉熵损失（分类问题）示例
# 形状：(样本数,)，每个元素是类别编号（0,1,2...）
y_true = torch.tensor([0, 1, 2], dtype=torch.int64)  # 正确格式：类别索引
# 模型输出的原始分数（logits），未经过softmax
# 形状：(样本数, 类别数)，3个样本，3个类别
y_predict = torch.tensor([[18.0, 9.0, 10.0], 
                          [2.0, 14.0, 6.0], 
                          [3.0, 8.0, 16.0]], dtype=torch.float32)
# 实例化交叉熵损失函数（内部自动对y_predict应用softmax）
criterion = nn.CrossEntropyLoss()
# 计算损失（使用类别索引标签）
loss = criterion(y_predict, y_true)
print(f"交叉熵损失值：{loss.item()}")

# 二元交叉熵损失（二分类问题）示例
y_true = torch.tensor([0, 1, 0, 1], dtype=torch.float32)  # 真实标签（0或1）
y_predict = torch.tensor([0.1, 0.9, 0.2, 0.8], dtype=torch.float32)  # 预测概率
loss = nn.BCELoss()  # 二元交叉熵损失
print("二元交叉熵损失：", loss(y_predict, y_true))

# 回归损失示例
y_true = torch.tensor([2.0, 3.0, 1.0], dtype=torch.float32)  # 真实值
y_predict = torch.tensor([1.0, 5.0, 4.0], dtype=torch.float32)  # 预测值
loss = nn.L1Loss()  # L1损失（绝对值误差）
loss = nn.MSELoss()  # MSE损失（均方误差）
loss = nn.SmoothL1Loss()  # Smooth L1损失（结合L1和MSE的优点）
```

## 06.指数加权平均

给予近期数据更高的权重，远期数据较低的权重，通过这种加权方式来计算平均值，从而在一定程度上反映数据的趋势
$$
v_t=β⋅v_{t−1}+(1−β)⋅x_t
$$

## 07.梯度下降优化

梯度下降优化算法中，可能会碰到以下情况：

1. 碰到**平缓区域**，梯度值较小，参数优化变慢
2. 碰到 **“鞍点”**，梯度为 0，参数无法优化

对于这些问题，出现了一些对梯度下降算法的优化方法，例如：**Momentum、AdaGrad、RMSprop、Adam** 等

1. **Momentum**：参数更新不仅依赖当前的梯度，还会参考之前累积的梯度 “动量”，梯度计算公式：$D_t = \beta \cdot S_{t-1} + (1 - \beta) \cdot W_t$
2. **AdaGrad**核心思想是**为不同参数动态调整学习率**—— 对频繁更新的参数使用较小的学习率，对稀疏更新的参数使用较大的学习率：$w_{t+1} = w_t - \frac{\eta}{\sqrt{r_t + \epsilon}} \cdot g_t$，非常适合处理稀疏数据
3. **RMSprop**主要用于解决 AdaGrad 学习率随时间单调递减的问题，**用指数加权平均替代累积和**，避免学习率过快衰减。$w_{t+1} = w_t - \frac{\eta}{\sqrt{s_t + \epsilon}} \cdot g_t$，$s_t = \beta \cdot s_{t-1} + (1 - \beta) \cdot g_t^2$
4. **Adam**结合了 Momentum（动量）算法和 RMSProp 算法的优点，自适应地调整每个参数的学习率，在许多深度学习任务中表现出色。

```python
import torch

# 初始化参数w（需要计算梯度）
w = torch.tensor([1.0], requires_grad=True, dtype=torch.float32)
# 定义损失函数：0.5*w²（示例损失）
loss = ((w**2) * 0.5).sum()

# 多种优化器示例（切换注释查看不同优化器效果）
optimizer = torch.optim.SGD([w], lr=0.01, momentum=0.9)  # SGD优化器（带动量）
optimizer = torch.optim.Adagrad([w], lr=0.01)  # Adagrad优化器
optimizer = torch.optim.RMSprop([w], lr=0.01, alpha=0.9)  # RMSprop优化器
optimizer = torch.optim.Adam([w], lr=0.01, betas=[0.9, 0.99])  # Adam优化器

# 第一次参数更新
optimizer.zero_grad()  # 清空梯度
loss.backward()        # 反向传播计算梯度
optimizer.step()       # 更新参数
print("第一次更新后梯度：", w.grad)
print("第一次更新后参数w：", w.detach())

# 第二次参数更新（重新计算损失）
loss = ((w**2) * 0.5).sum()
optimizer.zero_grad()
loss.backward()
optimizer.step()
print("第二次更新后梯度：", w.grad)
print("第二次更新后参数w：", w.detach()) # 将张量 w 从计算图中分离出来，使其不再参与梯度计算
```

## 08.学习率衰减

```python
y = torch.tensor([1.0])  # 目标输出

optimizer = torch.optim.SGD([w], lr=lr0, momentum=0.9)

# 学习率衰减策略（切换注释查看不同策略）
# 固定步长衰减
scheduler_lr = torch.optim.lr_scheduler.StepLR(optimizer, step_size=20, gamma=0.8) 
# 多步长衰减
scheduler_lr = torch.optim.lr_scheduler.MultiStepLR(optimizer, milestones=[20,60,90,135,180], gamma=0.8)  
scheduler_lr = torch.optim.lr_scheduler.ExponentialLR(optimizer, gamma=0.99)  # 指数衰减

lr_list = []     # 学习率列表

    # 每轮迭代更新参数
    for i in range(iter):
        # 计算损失：0.5*(w*x - y)^2
        loss = ((w * x - y) ** 2) * 0.5
        optimizer.zero_grad()  # 清空梯度
        loss.backward()        # 反向传播
        optimizer.step()       # 更新参数
    
    scheduler_lr.step()  # 更新学习率

# 绘制学习率随轮次变化曲线
```

**DropOut：在训练过程中**，Dropout 的实现是让神经元以超参数 p 的概率停止工作或者激活被置为 0，为了**保证训练和测试阶段神经元输出的期望一致，对未被置为 0 的进行缩放，输出缩放比例为 1/(1-p)**。训练过程可以认为是对完整的神经网络的一些子集进行训练，每次基于输入数据只更新子网络的参数 。

```python
def load_and_preprocess_data():
    """加载数据并完成预处理（标准化、格式转换等）"""

    data = pd.read_csv(DATA_PATH)

    y_test = torch.tensor(y_test, dtype=torch.int64)

    # 创建数据集和数据加载器
    train_dataset = TensorDataset(x_train, y_train)
    test_dataset = TensorDataset(x_test, y_test)
    train_loader = DataLoader(train_dataset, batch_size=BATCH_SIZE, shuffle=True)
    test_loader = DataLoader(test_dataset, batch_size=BATCH_SIZE, shuffle=False)
#                                     计算测试集长度使用
    return train_loader, test_loader, test_dataset, x_train.shape[1]  # 返回特征维度


# ----------------------------
# 3. 模型定义（简洁清晰）
# ----------------------------
class PricePredictionNet(nn.Module):
    def __init__(self, input_dim, dropout_rate=DROPOUT_RATE):
        super(PricePredictionNet, self).__init__()
        self.fc1 = nn.Linear(input_dim, 64)   # 第一层全连接
        self.fc2 = nn.Linear(64, 32)          # 第二层全连接
        self.out = nn.Linear(32, 4)           # 输出层（4个类别）
        self.dropout = nn.Dropout(dropout_rate)  # Dropout层（参数化，便于调优）
        self.relu = nn.ReLU()                # ReLU激活函数（复用）

    def forward(self, x):
        x = self.fc1(x)
        x = self.dropout(x)
        x = self.relu(x)

        x = self.fc2(x)
        x = self.dropout(x)
        x = self.relu(x)

        x = self.out(x)  # 输出层不激活（CrossEntropyLoss自带softmax）
        return x


# ----------------------------
# 4. 训练函数（优化逻辑，增加日志）
# ----------------------------
def train_model(model, train_loader, criterion, optimizer, epochs=EPOCHS):
    """训练模型并返回训练损失记录"""
    model.train()  # 开启训练模式（Dropout生效）
    train_losses = []  # 记录每轮损失

    for epoch in range(epochs):
        total_loss = 0.0  # 累计本轮总损失
        total_batches = 0  # 累计本轮批次数量

        for x_batch, y_batch in train_loader:
            # 前向传播
            y_pred = model(x_batch)
            loss = criterion(y_pred, y_batch)

            # 反向传播与参数更新
            optimizer.zero_grad()  # 梯度清零
            loss.backward()        # 计算梯度
            optimizer.step()       # 更新参数

            # 累计损失
            total_loss += loss.item()
            total_batches += 1

        # 计算本轮平均损失
        avg_loss = total_loss / total_batches
        train_losses.append(avg_loss)

        # 打印训练日志（每10轮打印一次，避免冗余）
        if (epoch + 1) % 10 == 0:
            print(f"Epoch [{epoch+1:3d}/{epochs}], Average Loss: {avg_loss:.4f}")

    # 绘制损失曲线
    plt.figure(figsize=(8, 5))
    plt.plot(range(1, epochs+1), train_losses, label='Training Loss')
    plt.xlabel('Epoch')
    plt.ylabel('Average Loss')
    plt.title('Training Loss Curve')
    plt.legend()
    plt.grid(alpha=0.3)
    plt.show()

    # 保存模型
    torch.save(model.state_dict(), MODEL_SAVE_PATH)
    print(f"模型已保存至：{MODEL_SAVE_PATH}")

    return model

# 5. 测试函数（优化评估方式，增加可读性）
# ----------------------------
def evaluate_model(model, test_loader, test_dataset):
    """评估模型在测试集上的准确率"""
    model.eval()  # 开启评估模式（关闭Dropout）
    correct = 0   # 正确预测的样本数

    with torch.no_grad():  # 关闭梯度计算，加速推理
        for x_batch, y_batch in test_loader:
            y_pred = model(x_batch)
            y_pred_class = torch.argmax(y_pred, dim=1)  # 取预测概率最大的类别
            correct += (y_pred_class == y_batch).sum().item()  # 累加正确数

    # 计算准确率（保留4位小数，转换为百分比）
    accuracy = correct / len(test_dataset)
    print(f"测试集准确率：{accuracy:.4f} ({accuracy*100:.2f}%)")
    return accuracy


# ----------------------------
# 6. 主函数（串联整个流程）
```

## 02.卷积层

```python
import torch.nn as nn
```

```python
MODEL_PATH = "model.pth"

def get_dataloaders():

class ImageClassifier(nn.Module):
    def __init__(self):
        super(ImageClassifier, self).__init__()
        # 卷积层 + 池化层
        self.conv1 = nn.Conv2d(3, 6, kernel_size=3, stride=1, padding=0)
        self.pool1 = nn.MaxPool2d(kernel_size=2, stride=2)
        self.conv2 = nn.Conv2d(6, 16, kernel_size=3, stride=1, padding=0)
        self.pool2 = nn.MaxPool2d(kernel_size=2, stride=2)
        
        # 全连接层
        self.fc1 = nn.Linear(16 * 6 * 6, 120)  # 16*6*6 = 576
        self.fc2 = nn.Linear(120, 60)
        self.out = nn.Linear(60, 10)
        
        # 激活函数（单独定义，避免重复创建）
        self.relu = nn.ReLU()
        self.leaky_relu = nn.LeakyReLU()

    def forward(self, x):
        # 卷积块 1
        x = self.leaky_relu(self.conv1(x))  # 保持原 LeakyReLU
        x = self.pool1(x)
        
        # 卷积块 2
        x = self.relu(self.conv2(x))  # 保持原 ReLU
        x = self.pool2(x)
        
        # 展平（替代 torch.reshape，更直观）
        x = x.view(x.size(0), -1)  # (batch, 16, 6, 6) → (batch, 576)
        
        # 全连接层
        x = self.relu(self.fc1(x))
        x = self.relu(self.fc2(x))
        
        # 输出层（CrossEntropyLoss 不需要手动 softmax）
        x = self.out(x)
        return x

        running_loss = 0.0
        total_samples = 0
        
        for batch_idx, (inputs, labels) in enumerate(train_loader):
            # 前向传播
            outputs = model(inputs)
            loss = criterion(outputs, labels)
            
            # 反向传播
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            
            # 累计损失与样本数
            running_loss += loss.item() * inputs.size(0)  # 按样本数加权累加
            total_samples += inputs.size(0)
        
        # 计算本轮平均损失
        epoch_loss = running_loss / total_samples
        loss_history.append(epoch_loss)
        
        # 打印训练日志（带进度条更直观）
        print(f"Epoch [{epoch+1:2d}/{epochs}]", 
              f"Loss: {epoch_loss:.4f}")
    
    return loss_history


# ======================== 5. 测试函数（修复路径与精度问题） ========================
def test_model(model, test_loader):
    """测试模型准确率"""
    model.eval()
    correct = 0
    total = 0
    
    with torch.no_grad():
        for inputs, labels in test_loader:
            outputs = model(inputs)
            _, predicted = torch.max(outputs.data, 1)  # 取预测概率最大的类别
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
    
    # 计算准确率（避免除零，且格式清晰）
    accuracy = correct / total if total != 0 else 0.0
    print(f"测试集准确率: {accuracy * 100:.2f}%")
    return accuracy


# ======================== 6. 主流程（解耦初始化与执行） ========================
if __name__ == "__main__":
    # 1. 初始化数据集
    train_loader, test_loader = get_dataloaders()
    
    # 2. 初始化模型、损失函数、优化器
    model = ImageClassifier()
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=LEARNING_RATE, betas=(0.9, 0.99))
    
    # 3. 查看模型结构（summary 移到主流程）
    summary(model, input_size=(3, 32, 32), batch_size=1)
```

## 01.词嵌入层

**词嵌入层的作用就是将文本转换为向量。**在传统的文本处理中，文本中的词通常以独热编码等方式表示，这种表示方法会导致维度灾难，并且无法体现词与词之间的语义关系 。而词嵌入层能够将词映射为低维、稠密的实值向量，比如将一个词表示成一个长度为 100、200 等的向量 。

词嵌入层首先会根据输入的词的数量构建一个**词向量矩阵**，例如：我们有 100 个词，每个词希望转换成 128 维度的向量，那么构建的矩阵形状即为: 100*128，输入的每个词都对应了一个该矩阵中的一个向量。

词嵌入层的实现基于一个预训练的词向量矩阵。在训练模型时，词嵌入层的参数（即词向量矩阵中的值）会随着模型的训练不断更新优化，使得语义相近的词在向量空间中的距离较近。比如， “香蕉” 和 “苹果” 这两个语义相近的词，它们对应的词向量在空间中的余弦距离会比较小。

```python
import jieba
import torch
import torch.nn as nn
from typing import List, Dict, Tuple

def process_text(text: str) -> Tuple[List[str], List[str], Dict[str, int]]:
    """
    处理文本：分词、去重并建立词到索引的映射
    Args:
        text: 待处理的中文文本      
    Returns:
        分词结果列表、去重词语列表、词语到索引的映射字典
    """
    # 分词处理
    words = jieba.lcut(text)
    
    # 去重并保持一定顺序（不同于set的完全无序）
    un_words = []
    seen = set()
    for word in words:
        if word not in seen:
            seen.add(word)
            un_words.append(word)
    
    # 建立词到索引的映射
    word_to_idx = {word: idx for idx, word in enumerate(un_words)}
    
    return words, un_words, word_to_idx

def create_embedding_layer(vocab_size: int, embedding_dim: int = 3) -> nn.Embedding:
    """创建词嵌入层"""
    return nn.Embedding(num_embeddings=vocab_size, embedding_dim=embedding_dim)

def get_word_embeddings(embedding_layer: nn.Embedding, 
                       words: List[str], 
                       word_to_idx: Dict[str, int]) -> Dict[str, torch.Tensor]:
    """
    获取词语的嵌入向量
    Args:
        embedding_layer: 词嵌入层
        words: 词语列表
        word_to_idx: 词语到索引的映射     
    Returns:
        词语到其嵌入向量的映射字典
    """
    embeddings = {}
    for word in words:
        idx = word_to_idx[word]
        # 使用no_grad()避免计算梯度，节省资源
        with torch.no_grad():
            embeddings[word] = embedding_layer(torch.tensor(idx))
    return embeddings

if __name__ == "__main__":
    # 示例文本
    text = "北京冬奥的进度条已经过半，不少外国运动员在完成自己的比赛后踏上归途。"
    
    # 处理文本
    words, un_words, word_to_idx = process_text(text)
    
    # 创建词嵌入层
    embedding_layer = create_embedding_layer(vocab_size=len(un_words))
    
    # 获取嵌入向量
    word_embeddings = get_word_embeddings(embedding_layer, un_words, word_to_idx)
    
    for word, vec in embeddings.items():
        print(f"词语: {word}")
        print(f"嵌入向量: {vec.numpy()}")  # 转换为numpy数组更易读
        print("-" * 30)
```

**RNN是如何计算的呢？**

![1753787463258](C:\Users\17630\AppData\Roaming\Typora\typora-user-images\1753787463258.png)

**h 表示隐藏状态，每一次的输入都会包含两个值：上一个时间步的隐藏状态、当前状态的输入值，输出当前时间步的隐藏状态和当前时间步的预测结果**。

**每个神经元内部是如何计算的呢？**
$$
h_t = \tanh\left(W_{ih} x_t + b_{ih} + W_{hh} h_{(t-1)} + b_{hh}\right)
$$

## 03.文本生成

**RNN 的输入、输出和隐藏状态**

假设我们使用 **`batch_first=True`**（推荐，更符合直观思维），定义以下参数：

- `batch_size`：批次大小（一次输入的样本数）
- `seq_len`：序列长度（每个样本包含的时间步数量）
- `input_size`：输入特征维度（词嵌入后的维度）
- `hidden_size`：RNN 隐藏层维度
- `num_layers`：RNN 的层数
- `vocab_size`：词汇表大小（输出类别数）

**1. 输入 (`x`)**

- **形状**：`(batch_size, seq_len)`
  这是经过词嵌入前的输入（每个元素是词的索引）。经过嵌入层（`nn.Embedding`）后，形状变为：
  `(batch_size, seq_len, input_size)`

**2. 隐藏状态 (`h`)**

- **隐藏状态 (h0)**：`(num_layers, batch_size, hidden_size)`

**3. RNN 输出 (`out`)**

```
(batch_size, seq_len, hidden_size)
```

- 后续需要通过全连接层（`nn.Linear`）映射到词汇表大小：`(batch_size, seq_len, vocab_size)`

**周杰伦歌词生成任务：**

**构建词表**：所谓的词表就是将数据进行分词，然后给每一个词分配一个唯一的编号，便于我们送入词嵌入层获取每个词的词向量。

```python
import torch
from jieba import lcut
from torch import nn
from torch.utils.data import Dataset, DataLoader


# 构建词汇表
def build_vocab():
    all_words = []
    unwords = []
    for line in open('data/jaychou_lyrics.txt', 'r', encoding='utf-8'):
        line = line.strip()  # 去除首尾空白
        words = lcut(line)
        all_words.extend(words)  # 使用extend而不是append
        for word in words:
            if word not in unwords:
                unwords.append(word)
    # 添加一个特殊的终止符
    if ' ' not in unwords:
        unwords.append(' ')
    word2idx = {word: idx for idx, word in enumerate(unwords)}
    all_words_idx = [word2idx[word] for word in all_words if word in word2idx]
    return unwords, word2idx, len(unwords), all_words_idx


# 数据集类（修复了继承问题）
class JayDataset(Dataset):  # 类名应大写，且正确继承Dataset
    def __init__(self, all_words_idx, num_char=10):  # 默认序列长度为10
        self.all_words_idx = all_words_idx
        self.num_char = num_char
        self.word_count = len(all_words_idx)
        # 计算有效样本数
        self.num = max(0, self.word_count - self.num_char)

    def __len__(self):
        return self.num

    def __getitem__(self, idx):
        if idx >= self.num:
            raise IndexError("索引超出范围")
        x = self.all_words_idx[idx:idx + self.num_char]
        y = self.all_words_idx[idx + 1:idx + self.num_char + 1]
        return torch.tensor(x), torch.tensor(y)


# 模型类（修复了RNN处理和隐藏状态初始化）
class Net(nn.Module):
    def __init__(self, vocab_size):  # 参数名应清晰
        super(Net, self).__init__()
        self.embedding = nn.Embedding(num_embeddings=vocab_size, embedding_dim=128)
        self.rnn = nn.RNN(
            input_size=128,
            hidden_size=256,
            num_layers=2,
            batch_first=True  # 重要：使用batch_first=True更符合习惯
        )
        self.out = nn.Linear(in_features=256, out_features=vocab_size)

    def forward(self, x, h):
        x = self.embedding(x)  # (batch_size, seq_len) -> (batch_size, seq_len, input_size)
        x, h = self.rnn(x, h)  # 由于batch_first=True，不需要转置
        out = self.out(x)  # (batch_size, seq_len, hidden_size) -> (batch_size, seq_len, vocab_size)
        return out, h

    def init_hidden(self, bs):
        # 隐藏状态的形状：(num_layers, batch_size, hidden_size)
        return torch.zeros(self.rnn.num_layers, bs, self.rnn.hidden_size)


# 训练函数
def train(dataset, model):
    model.train()
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
    dataloader = DataLoader(dataset, batch_size=32, shuffle=True)

    for epoch in range(100):
        total_loss = 0
        for x, y in dataloader:
            # 初始化隐藏状态
            h0 = model.init_hidden(x.size(0))
            # 前向传播，输出 out 的形状是 (batch_size, seq_len, vocab_size)
            out, _ = model(x, h0)
            # 调整输出和目标形状以匹配损失函数
            out = out.reshape(-1, out.size(-1))  # (batch_size*seq_len, vocab_size)
            y = y.reshape(-1)  # (batch_size*seq_len,)
            # 计算损失
            loss = criterion(out, y)
            # 反向传播和优化
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            total_loss += loss.item()

        if (epoch + 1) % 10 == 0:
            print(f"Epoch {epoch + 1}, 平均损失: {total_loss / len(dataloader):.4f}")

    # 保存模型参数
    torch.save(model.state_dict(), 'model.pth')
    print("模型已保存")


# 预测函数
def predict(model, start_word, length, unwords, word2idx):
    model.load_state_dict(torch.load('model.pth'))
    model.eval()

    word_index = word2idx[start_word]
    h0 = model.init_hidden(1)  # 预测时batch_size为1
    words_list = [start_word]

    with torch.no_grad():  # 预测时不需要计算梯度
        current_input = torch.tensor([[word_index]])  # 形状: (1, 1)

        for _ in range(length - 1):
            out, h0 = model(current_input, h0)
            # 取最后一个时间步的输出
            pred_probs = out[0, -1, :]
            word_index = torch.argmax(pred_probs).item()
            words_list.append(unwords[word_index])
            # 更新输入
            current_input = torch.tensor([[word_index]])

    return ''.join(words_list)


if __name__ == '__main__':
    # 构建词表
    unwords, word2idx, vocab_size, all_words_idx = build_vocab()

    # 创建数据集（每个样本长度10）
    dataset = JayDataset(all_words_idx, num_char=10)

    # 初始化模型
    model = Net(vocab_size)
    # 训练模型
    train(dataset, model)

    # 预测：从"青春"开始生成50个词
    result = predict(model, '青春', 50, unwords, word2idx)
    print(result)
```
