# 数据结构与算法

| **时间复杂度**    | **对应算法/数据结构典型案例**                                |
| ----------------- | ------------------------------------------------------------ |
| **$O(1)$**        | 访问数组索引、入栈/出栈、入队/出队、哈希表单次查找/插入、获取链表头节点。 |
| **$O(\log N)$**   | **二分查找**、二叉搜索树（BST）查找、向上/向下过滤（堆调整过程）、求幂运算（快速幂）。 |
| **$O(N)$**        | 遍历数组/链表、线性查找、**Quick Select（快速选择第 K 大）**、求最大值/最小值、计数排序、基数排序。 |
| **$O(N \log N)$** | **快速排序**、**堆排序**、**归并排序**、`std::sort`（内省排序）、对 $N$ 个元素进行二分查找。 |
| **$O(N^2)$**      | 冒泡排序、选择排序、插入排序、**双重嵌套循环遍历**（如：二维数组初始化、暴力破解两数之和）。 |
| **$O(2^N)$**      | **递归计算斐波那契数列**、求解幂集（所有子集）、递归处理汉诺塔问题。 |
| **$O(N!)$**       | **全排列算法**（Permutations）、旅行商问题（TSP）的暴力解法。 |

```c++
class Array{
public:    
    Array(int size = 10):mCap(size),mCur(0){
        mpArr = new int[mCap]();//初始化为0
    }
    ~Array(){
        delete[]mpArr;
        mpArr = nullptr;
    }
    void push_back(int val){
        if(mCur == mCap){
            expand(mCap*2);
        }
        mpArr[mCur++] = val;
    }
    void pop_back(){
        if(mCur == 0){
            return;
        }
        mCur--;
    }
    void insert(int pos,int val){
        if(pos<0 || pos>mCur){
			return;
        }
        if(mCur == mCap){
            expand(mCap*2);
        }
        for(int i = mCur-1;i >= pos;i--){
            mpArr[i+1] = mpArr[i];
        }
        mpArr[pos] = val;
        mCur++;
    }
    void erase(int pos){//按位置删除
        if(pos<0 || pos>=mCur){
            return;
        }
        for(int i = pos+1;i < mCur;i++){
            mpArr[i-1] = mpArr[i];
        }
        mCur--;
    }
    int find(int val){//查找元素返回下标
        for(int i = 0;i < mCur;i++){
            if(mpArr[i] == val){
                return i;
            }
        }
        return -1;
    }
private:
    int* mpArr;//指向可扩容的数组内存
    int mCap;//数组容量
    int mCur;//数组有效元素个数
    void expand(int size){
        int* p = new int[size];
        memcpy(p,mpArr,sizeof(int)*mCap);
        delete []mpArr;
        mpArr = p;
        mCap = size;
    }
}

int main(){
    Array arr;
    for(int i = 0;i < 20;i++){
        arr.push_back(rand()%100);
    }
    
    return 0;
}
```

```c++
//整形数组，把偶数调整到数组的左边，奇数调整到数组的右边
void AdjustArray(int arr[], int size) {//参考快排的partition函数
    int left = 0;             
    int right = size - 1;     

    while (left < right) {
        // 1. 从左向右找到第一个奇数 (奇数 % 2 != 0)
        // 如果是偶数就继续右移
        while (left < right && arr[left] % 2 == 0) {
            left++;
        }
        // 2. 从右向左找到第一个偶数 (偶数 % 2 == 0)
        // 如果是奇数就继续左移
        while (left < right && arr[right] % 2 != 0) {
            right--;
        }
        // 3. 交换这两个“放错位置”的数
        if (left < right) {
            int temp = arr[left];
            arr[left] = arr[right];
            arr[right] = temp;            
            // 交换后，两下标各向中间进一位
            left++;
            right--;
        }
    }
}
```

```c++
//给你一个数组nums和一个值val，你需要原地移除所有数值等于val的元素，并返回移除后数组的长度
int removeElement(int nums[], int size, int val) {
    int slow = 0; // 慢指针：指向下一个待填充的位置

    for (int fast = 0; fast < size; fast++) {
        // 如果当前快指针指向的值不等于 val
        if (nums[fast] != val) {
            // 将快指针的值赋给慢指针位置
            nums[slow] = nums[fast];
            // 慢指针向前移动
            slow++;
        }
    } 
    // 循环结束后，slow 的值正好就是新数组的长度
    return slow;
}
```

```c++
struct Node {
    int data_;
    Node* next_;
    Node(int data = 0) : data_(data), next_(nullptr) {}
};

class Clink {
public:
    Clink() : mSize(0) {    // 构造函数：初始化头节点
        head_ = new Node(); 
    }
    ~Clink() {    // 析构函数：释放所有节点
        clear();
        delete head_;
    }
    void push_front(int val) {    // 头部插入 O(1)
        Node* newNode = new Node(val);
        newNode->next_ = head_->next_;
        head_->next_ = newNode;
        mSize++;
    }
    void push_back(int val) {    // 尾部插入 O(n)
        Node* p = head_;
        while (p->next_ != nullptr) {
            p = p->next_;
        }
        p->next_ = new Node(val);
        mSize++;
    }
    void insert(int pos, int val) {    // 在指定位置插入（pos从0开始） O(n)
        if (pos < 0 || pos > mSize) return;      
        Node* p = head_;
        for (int i = 0; i < pos; ++i) { // 找到待插入位置的前驱
            p = p->next_;
        }
        Node* newNode = new Node(val);
        newNode->next_ = p->next_;
        p->next_ = newNode;
        mSize++;
    }
    void pop_front() {    // 头部删除 O(1)
        if (head_->next_ == nullptr) return;
        Node* tmp = head_->next_;
        head_->next_ = tmp->next_;
        delete tmp;
        mSize--;
    }
    void pop_back() {    // 尾部删除 O(n)
        if (head_->next_ == nullptr) return;
        Node* prev = head_;
        Node* p = head_->next_;
        while (p->next_ != nullptr) { // 找到最后一个节点和它的前驱
            prev = p;
            p = p->next_;
        }
        delete p;
        prev->next_ = nullptr;
        mSize--;
    }
    void erase(int pos) {    // 删除指定位置的元素 O(n)
        if (pos < 0 || pos >= mSize) return;
        Node* prev = head_;
        for (int i = 0; i < pos; ++i) { // 找到要删节点的前驱
            prev = prev->next_;
        }
        Node* target = prev->next_;
        prev->next_ = target->next_;
        delete target;
        mSize--;
    }
    void clear() {    // 清空链表（保留头节点）
        while (head_->next_ != nullptr) {
            pop_front();
        }
    }
    int size() const { return mSize; }
    bool empty() const { return mSize == 0; }
    void show() const {    // 打印链表（方便调试）
        Node* p = head_->next_;
        while (p) {
            cout << p->data_ << " -> ";
            p = p->next_;
        }
        cout << "NULL" << endl;
    }
private:
    Node* head_;
    int mSize; // 记录有效元素个数
};
```

```c++
//求链表的倒数第k个节点
Node* getKthFromEnd(Node* head, int k) {
    // 基础检查：如果链表为空或 k 不合法
    if (head == nullptr || k <= 0) return nullptr;
    
    Node* slow = head;
    Node* fast = head;
    // 1. 快指针先走 k 步
    for (int i = 0; i < k; i++) {
        // 关键点：如果 k 大于链表长度，这里会发生空指针解引用
        if (fast == nullptr) return nullptr; 
        fast = fast->next;
    }
    // 2. 两个指针同步走
    while (fast != nullptr) {
        slow = slow->next;
        fast = fast->next;
    }
    // 此时 slow 所在的位置就是倒数第 k 个节点
    return slow;
}
```

```c++
//合并两个有序链表
Node* mergeList(Node* head1, Node* head2) {
    if (head1 == nullptr) return head2;
    if (head2 == nullptr) return head1;

    Node* dummy = new Node(); // 虚拟头节点，用来接住合并后的第一个节点
    Node* cur = dummy;
    // 只要两个链表都没走到头
    while (head1 != nullptr && head2 != nullptr) {
        if (head1->data_ <= head2->data_) {
            cur->next = head1;    // 直接把 cur 指向已有的节点
            head1 = head1->next;  // 移动 head1
        } else {
            cur->next = head2;    // 直接把 cur 指向已有的节点
            head2 = head2->next;  // 移动 head2
        }
        cur = cur->next;          // cur 向前推进
    }
    // 只要有一个链表走完了，直接把另一个链表剩下的部分接上去
    cur->next = (head1 != nullptr) ? head1 : head2;

    Node* res = dummy->next;
    delete dummy; // 释放虚拟头节点（它是唯一 new 出来的）
    return res;
}
```

```c++
bool isLoop(Node* head) {
    Node* slow = head;
    Node* fast = head;
    // 必须确保 fast 和 fast->next 都不为空，才能走两步
    while (fast != nullptr && fast->next != nullptr) {

Node* getNode(Node* head) {
    if (head == nullptr || head->next == nullptr) return nullptr;

    Node* slow = head;
    Node* fast = head;

    while (fast != nullptr && fast->next != nullptr) {
        slow = slow->next;
        fast = fast->next->next;        
        if (slow == fast) { // 相遇了
            Node* node = head;
            while (node != slow) { // slow 或 fast 都可以
                node = node->next;
                slow = slow->next;
            }
            return node; // 再次相遇点即为入口
        }
    }
    return nullptr;
}
```

```c++
    if(head1 == nullptr || head2 == nullptr) return false;
    int len1 = 0;
    int len2 = 0;
    Node* first = head1;
    Node* second = head2;

        len1++;
        first = first->next;
    }

        len2++;
        second = second->next;

        if(head1 == head2){
            return true;
        }
        head1 = head1->next;
        head2 = head2->next;
    }
    return false;
}
```

```c++
//删除链表倒数第N个节点
Node* remove(Node* head, int N) {
    Node* dummy = new Node();
    dummy->next = head;
    Node* slow = dummy;
    Node* fast = dummy;
    // 快指针走 N+1 步。注意：这里不直接 return head
    for (int i = 0; i < N + 1; i++) {
        if (fast == nullptr) return head; // 只有当 N 超过长度时才返回
        fast = fast->next;
    }

    while (fast != nullptr) {
        slow = slow->next;
        fast = fast->next;
    }

    Node* nodeToDelete = slow->next;
    slow->next = nodeToDelete->next; // 此时 nodeToDelete 一定存在
    delete nodeToDelete;

    Node* res = dummy->next; // 获取真实的新头节点
    delete dummy;
    return res;
}
```

```c++
//给你一个链表的头结点head，旋转链表，将链表每个节点向右移动k个位置
Node* rotateList(Node* head, int k) {
    // 1. 基础检查
    if (head == nullptr || head->next == nullptr || k == 0) return head;
    // 2. 计算长度
    int len = 0;
    Node* p = head;
    while (p != nullptr) {
        len++;
        p = p->next;
    }
    // 3. 取模：如果 k 是长度倍数，旋转后还是原样
    k = k % len;
    if (k == 0) return head;
    // 4. 找到“切断点”：即新链表的尾部
    // 向右移动 k 个，意味着前 len - k 个节点留在前面，最后一个是新尾
    p = head;
    for (int i = 0; i < len - k - 1; i++) {
        p = p->next;
    }
    // 5. 切断并拼接
    Node* newhead = p->next; // 找到新头
    p->next = nullptr;       // 断开，旧尾指向空
    // 6. 找到新段的尾部，连上旧头
    Node* temp = newhead;
    while (temp->next != nullptr) {
        temp = temp->next;
    }
    temp->next = head;

    return newhead;
}
```

```c++
//有 n 个人围成一圈，从第一个人开始报数，报到 m 的人出列，剩下的人继续从 1 开始报数，直到最后只剩一个人。
int josephus(int n, int m) {
    vector<int> people;
    for (int i = 1; i <= n; i++) people.push_back(i);

    int cur = 0; // 从第一个人开始
    while (people.size() > 1) {
        // 关键逻辑：计算下一个要删除的人的位置
        cur = (cur + m - 1) % people.size(); 
        people.erase(people.begin() + cur); // 物理删除
    }
    return people[0];
}
```

```c++
//顺序栈
class SeqStack {
public:
    SeqStack(int size = 10) : mtop(0), mcap(size) {
        mpStack = new int[mcap];
    }
    ~SeqStack() {
        delete[] mpStack;
        mpStack = nullptr;
    }
    // 入栈
    void push(int val) {
        if (mtop == mcap) {
            expand(); // 空间满了，触发扩容
        }
        mpStack[mtop++] = val;
    }
    // 出栈
    void pop() {
        if (isEmpty()) return;
        mtop--; // 逻辑删除，不需要物理清除
    }
    // 查看栈顶元素
    int top() const {
        if (isEmpty()) throw "Stack is empty!";
        return mpStack[mtop - 1];
    }
    bool isEmpty() const { return mtop == 0; }
    int size() const { return mtop; }
private:
    int* mpStack;
    int mtop; 
    int mcap;
    // 动态扩容逻辑
    void expand() {
        int newCap = mcap * 2;
        int* pTmp = new int[newCap];
        for (int i = 0; i < mcap; i++) {
            pTmp[i] = mpStack[i];
        }
        delete[] mpStack;
        mpStack = pTmp;
        mcap = newCap;
    }
};
```

```c++
//逆波兰表达式是一种后缀表达式，就是运算符写在后面，例如1 2 + 3 4 + *
int evalRPN(vector<string>& tokens) {
    stack<int> st;
    for (string& s : tokens) {
        // 判断是运算符还是数字
        if (s == "+" || s == "-" || s == "*" || s == "/") {
            int r = st.top(); st.pop();
            int l = st.top(); st.pop();
            
            if (s == "+") st.push(l + r);
            else if (s == "-") st.push(l - r);
            else if (s == "*") st.push(l * r);
            else if (s == "/") st.push(l / r);
        } else {
            st.push(stoi(s));  // 是数字（包括多位数和负数）,stoi可以直接转string为int
        }
    }
    return st.top();
}
```

**中缀转后缀(逆波兰)表达式**

```c++
/*
在算法中，我们需要给符号定级：
*, / : 优先级 2
+, - : 优先级 1
( : 优先级 0（在栈外优先级最高，在栈内优先级最低，确保任何符号都能压在它上面）
 
1.数字：直接输出（添加到后缀表达式结果中）。
2.左括号 (：直接入栈。
3.右括号 )：不断弹出栈顶运算符并输出，直到遇到左括号 ( 为止。最后弹出 ( 但不输出。
4.运算符（+, -, *, /）：
	如果当前运算符优先级 大于 栈顶运算符优先级，直接入栈。
	否则（小于或等于），将栈顶运算符弹出并输出，重复此步骤，直到当前运算符优先级大于栈顶，或者	 栈为空，再将当前运算符入栈。
5.扫描结束：将栈中剩余的运算符全部弹出并输出。
*/
int getPriority(char op) {
    if (op == '*' || op == '/') return 2;
    if (op == '+' || op == '-') return 1;
    return 0;
}

string infixToPostfix(string s) {
    string res = "";
    stack<char> st;
    
    for (char ch : s) {
        if (isdigit(ch)) {
            res += ch; // 简单起见，处理个位数
        } else if (ch == '(') {
            st.push(ch);
        } else if (ch == ')') {
            while (!st.empty() && st.top() != '(') {
                res += st.top();
                st.pop();
            }
            st.pop(); // 弹出 '('
        } else {
            // 比较优先级
            while (!st.empty() && getPriority(st.top()) >= getPriority(ch)) {
                res += st.top();
                st.pop();
            }
            st.push(ch);
        }
    }
    // 最后清空栈
    while (!st.empty()) {
        res += st.top();
        st.pop();
    }
    return res;
}
```

```c++
        delete[] _data;
    }
    // 入队
    void push(int val) {
        if (isFull()) {
            _resize(_capacity * 2); // 空间不足，扩容
        }
        _data[_rear] = val;
        _rear = (_rear + 1) % _capacity;
    }
    // 出队
    void pop() {
        if (empty()) {
            throw "Queue is empty";
        }
        _front = (_front + 1) % _capacity;
    }
    // 获取队头元素
    int front() const {
        if (empty()) throw "Queue is empty";
        return _data[_front];
    }
    // 获取队尾元素
    int back() const {
        if (empty()) throw "Queue is empty";
        // rear 指向的是下一个插入位置，所以要减 1
        return _data[(_rear - 1 + _capacity) % _capacity];
    }
    // 判空
    bool empty() const {
        return _front == _rear;
    }
    // 当前元素个数
    size_t size() const {
        return (_rear - _front + _capacity) % _capacity;
    }
private:
    int* _data;
    int _capacity;
    int _front;
    int _rear;
    // 内部判满：为了在 push 时清晰调用
    bool isFull() const {
        return (_rear + 1) % _capacity == _front;
    }
    // 动态扩容
    void _resize(int new_capacity) {
        int* new_data = new int[new_capacity];
        int current_size = size();
        // 核心：按逻辑顺序将旧数据拷贝到新数组
        // 从旧的 front 开始，依次取出 current_size 个元素
        for (int i = 0; i < current_size; ++i) {
            new_data[i] = _data[(_front + i) % _capacity];
        }
        delete[] _data;
        _data = new_data;
        _capacity = new_capacity;        
        // 重置索引：新数组从 0 开始排列
        _front = 0;
        _rear = current_size;
    }
};
```

```c++
public:
    Queue() : _front(nullptr), _rear(nullptr), _size(0) {}

        }

        Node* newNode = new Node(val);

            _front = _rear = newNode;
        } else {
            _rear->next = newNode;
            _rear = newNode;

            throw "Queue is empty";

        _front = _front->next;
        delete temp;
        _size--;        
        // 如果删完后队列空了，rear 也需要置空

            _rear = nullptr;
        }

        return _front->data;

    // 获取队尾元素
    int back() const {
        if (empty()) throw "Queue is empty";

    bool empty() const {

    }
    // 当前大小

        return _size;
```

```c++
//使用两个栈实现队列，支持push、pop、front、empty
//s1 负责入队，s2 负责出队，只有当 s2 为空时，才一次性把 s1 的数据倒进 s2,均摊复杂度为O(1)
class Queue {
public:
    // 入队：直接进入输入栈
    void push(int val) {
        s1.push(val);
    }
    // 出队：从输出栈弹出
    void pop() {
        moveIfEmpty();
        if (s2.empty()) throw "Queue is empty";
        s2.pop();
    }
    // 获取队头：从输出栈查看
    int front() {
        moveIfEmpty();
        if (s2.empty()) throw "Queue is empty";
        return s2.top();
    }
    // 判空：两个栈都为空才算空
    bool empty() {
        return s1.empty() && s2.empty();
    }
private:
    stack<int> s1; // 输入栈 (Push Stack)
    stack<int> s2; // 输出栈 (Pop Stack)
    // 核心优化逻辑：
    // 只有当 s2 为空时，才将 s1 的全部元素倒入 s2。
    // 这样 s2 顶部的元素就是最早进入 s1 的元素（FIFO）。
    void moveIfEmpty() {
        if (s2.empty()) {
            while (!s1.empty()) {
                s2.push(s1.top());
                s1.pop();
            }
        }
    }
};
```

```c++
//搜索的过程实际上就是对一颗BST树从根节点进行搜索的过程，因此时间复杂度为O(logn)
int binarySearch(const vector<int>& nums, int target) {
    int left = 0;
    int right = nums.size() - 1;

    while (left <= right) {
        // 避免直接 (left + right) / 2 导致的整型溢出
        int mid = left + (right - left) / 2;

        if (nums[mid] == target) {
            return mid; // 找到目标，返回下标
        } else if (nums[mid] < target) {
            left = mid + 1; // 目标在右侧
        } else {
            right = mid - 1; // 目标在左侧
        }
    }
    return -1; // 未找到
}
```

```c++
//冒泡排序：通过相邻元素的比较和交换，使较大的元素逐渐“浮”到数组的末尾。
void bubbleSort(vector<int>& nums) {
    int n = nums.size();
    for (int i = 0; i < n - 1; ++i) {
        bool swapped = false; // 优化：记录本轮是否发生交换      
        // 每一轮 j 的终点都在收缩，因为末尾已经是排好序的了
        for (int j = 0; j < n - 1 - i; ++j) {
            if (nums[j] > nums[j + 1]) {
                swap(nums[j], nums[j + 1]);
                swapped = true;
            }
        }        
        // 如果没有交换，说明已经有序，直接退出
        if (!swapped) break;
    }
}
```

```c++
//选择排序：每一轮从待排序的数据中找出最小（或最大）的一个元素，存放在序列的起始位置，直到全部待排序的数据元素排完。
void selectionSort(vector<int>& arr) {
    int n = arr.size(); 
    // 移动未排序子数组的边界
    for (int i = 0; i < n; i++) {
        // 寻找 [i, n] 区间里的最小元素
        int min_idx = i;
        for (int j = i + 1; j < n; j++) {
            if (arr[j] < arr[min_idx]) {
                min_idx = j;
            }
        }   
        // 将找到的最小元素与第 i 个位置交换
        swap(arr[i], arr[min_idx]);
    }
}
```

#### ==插入排序==

```c++
//如果数据趋于有序，那么插入排序是所有排序算法中效率最高的排序算法！
//插入排序：假设第一个元素是已经排好序的，从下一个元素开始，在已排好序的序列中从右向左扫描，如果待排序元素小于扫描到的元素，就让扫描到的元素向右挪一个位置，直到找到合适的位置。
void insertionSort(vector<int>& arr) {
    int n = arr.size();

    for (int i = 1; i < n; ++i) {
        int key = arr[i]; // 当前要插入的“牌”
        int j = i - 1;
        // 将 key 与前面的有序序列对比，比 key 大的元素向右移动
        // 注意：j 是 int 类型，因为它可能减到 -1
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }
        // 找到位置，插入 key
        arr[j + 1] = key;
    }
}
```

```c++
//快速排序：每次选择基准数，把小于基准数的放到基准数的左边，大于基准数的放到基准数的右边，采用分治思想处理剩余的序列元素，直到整个序列变为有序

// 分区函数
int partition(vector<int>& arr, int low, int high) {
    // 简单起见，取最后一个元素作为基准
    int pivot = arr[high]; 
    int i = low - 1; // 🌟🌟🌟🌟🌟 i 是小于基准区域的右边界

    for (int j = low; j < high; j++) {
        // 如果当前元素小于或等于基准
        if (arr[j] <= pivot) {
            i++;
            swap(arr[i], arr[j]);
        }
    }
    // 最后把基准元素交换到中间
    swap(arr[i + 1], arr[high]);
    return i + 1; // 返回基准的最终位置
}
void quickSort(vector<int>& arr, int low, int high) {
    if (low < high) {
        // 获取分区点
        int pi = partition(arr, low, high);
        // 递归排序左半部分和右半部分
        quickSort(arr, low, pi - 1);
        quickSort(arr, pi + 1, high);
    }
}
/*
我们以数组 [3, 5, 2, 4] 为例，基准值（pivot）取最后一个元素 4。
1.初始状态: i 指向 low-1（起始位置左侧），j 从 low 开始扫描。
			[ 3,  5,  2,  4 ]
		  	  j           pivot
			i
2.j = 0, arr[0] = 3,由于 3 <= 4，符合条件。i 向右移一位，然后 swap(arr[i], arr[j])。 （因为 i 和 j 此时重合，交换后位置不变）
			[ 3,  5,  2,  4 ]
  			  i
 			  j           pivot
3.j = 1, arr[1] = 5，由于 5 > 4，不符合条件。i 不动，j 继续向右移。
			[ 3,  5,  2,  4 ]
			  i   j       pivot
4.j = 2, arr[2] = 2，由于 2 <= 4，符合条件。i 向右移一位（指向 5），然后 swap(arr[i], arr[j])。 （即交换 5 和 2）
			[ 3,  2,  5,  4 ]
     			  i   j   pivot
5.循环结束，基准归位，j 扫描到了 high 之前。现在执行 swap(arr[i + 1], arr[high])。 （即交换 5 和 4）
			[ 3,  2,  4,  5 ]
			          ^
			          pi (返回索引 2)
*/
```

#### ==堆排序==

**==二叉堆：就是一颗完全二叉树==，分为两种典型的堆，分别是大根堆和小根堆**

**如果arr[i] <= arr[2*i+1]&&arr[i] <=arr[2*i+2],就是==小根堆，即当前节点小于孩子节点==**

**如果arr[i] >= arr[2*i+1]&&arr[i] >=arr[2*i+2],就是==大根堆，即当前节点大于孩子节点==**

**当前节点的父节点下标：==(i-1)/2==**

```c++
//堆排序
// 维护堆的性质：对以 i 为根的子树进行调整
void heapify(vector<int>& arr, int n, int i) {
    int largest = i;       // 假设当前节点最大
    int left = 2 * i + 1;  // 左孩子
    int right = 2 * i + 2; // 右孩子
    // 如果左孩子比根大
    if (left < n && arr[left] > arr[largest])
        largest = left;
    // 如果右孩子比当前最大的还大
    if (right < n && arr[right] > arr[largest])
        largest = right;
    // 如果最大值不是根，则交换并递归调整
    if (largest != i) {
        swap(arr[i], arr[largest]);
        // 递归调整受影响的子树
        heapify(arr, n, largest);
    }
}

void heapSort(vector<int>& arr) {
    int n = arr.size();
    // 1. 建堆：从最后一个非叶子节点开始向上调整
    // 最后一个非叶子节点索引为 n/2 - 1
    for (int i = n / 2 - 1; i >= 0; i--) {
        heapify(arr, n, i);
    }
    // 2. 一个个从堆顶取出元素
    for (int i = n - 1; i > 0; i--) {
        // 将当前堆顶（最大值）移到末尾
        swap(arr[0], arr[i]);
        // 对缩小的堆再次进行调整
        heapify(arr, i, 0);//i = n-1为缩小后的堆
    }
}
/*
如果你想实现升序排序（从小到大），我们需要构建的是大根堆。
如果你想实现降序排序（从大到小），则构建小根堆。
为什么升序要用大根堆？
看代码中的第二步：我们每次把堆顶（最大的数）换到数组的最后面。
第一次换，最大的数到了 arr[n-1]；
第二次换，第二大的数到了 arr[n-2]；
以此类推，最后数组就变成了升序。
*/
```



#### ==归并排序==

```c++
//归并排序：将数组从中间切开，分成左右两半，直到每个子数组只有一个元素，然后递归地对两个子数组进行归并排序。

// 合并两个有序区间：[left, mid] 和 [mid + 1, right]
void merge(vector<int>& arr, int left, int mid, int right) {
    vector<int> temp;    // 创建临时容器存放合并结果

    int i = left;      // 左序列指针
    int j = mid + 1;   // 右序列指针
    // 比较两个序列，谁小就谁先入场
    while (i <= mid && j <= right) {
        if (arr[i] <= arr[j]) {
            temp.push_back(arr[i++]);
        } else {
            temp.push_back(arr[j++]);
        }
    }
    // 处理剩余元素
    while (i <= mid) temp.push_back(arr[i++]);
    while (j <= right) temp.push_back(arr[j++]);
    // 将临时容器里的有序数据写回原数组
    for (int k = 0; k < temp.size(); ++k) {
        arr[left + k] = temp[k];
    }
}

void mergeSort(vector<int>& arr, int left, int right) {
    if (left >= right) return;

    int mid = left + (right - left) / 2; // 防止溢出

    mergeSort(arr, left, mid);      // 递归左半部分
    mergeSort(arr, mid + 1, right); // 递归右半部分
    
    merge(arr, left, mid, right);   // 合并左右两部分
}
```

==**1️⃣STL里sort算法用的是什么排序算法？**==

算法开始时，**默认使用快速排序**。**当递归深度超过某个阈值（通常是 $2 \log_2 n$）时，说明快速排序可能陷入了不当的基准选择，此时算法会切换为堆排序**，堆排序在最坏情况下也能保证 $O(n \log n)$，当**分割后的子序列长度小于某个阈值（通常是 32）时会切换为插入排序**。

==2️⃣**快速排序的时间复杂度不是稳定的nlogn，最坏情况下会变成$n^2$（数组已经有序，并且基准取第一个或最后一个），怎么解决复杂度恶化问题？**==

**优化基准选择：随机化选择，三数取中法**（第一个，中间位置的，最后一个元素取中位数作为基准）

**==3️⃣快速排序递归实现时，怎么解决递归层次过深的问题？递归过深会引发什么问题==**

**使用内省排序std::sort**，在递归时记录当前的深度，如果深度超过阈值，转为堆排序

**优化基准选择**：随机化选择，三数取中法

**递归过深引发的问题：栈溢出**，递归函数调用会在栈上开辟栈帧，栈空间大小只有几M，如果递归层数过深，栈空间会被耗尽，系统会强制杀掉程序



### 哈希表

1️⃣ **==哈希函数==：将任意长度的输入（Key）映射为固定范围的整数（下标）-->除留余数法。**

2️⃣ **==哈希冲突==：不同的 Key 经过哈希函数计算后，得到了同一个下标。 这就是冲突。**

**处理冲突的两大门派⬇️**

==**A. 链地址法—— 经典的 C++ unordered_map 做法**==

**数组的每个位置不直接存数据，而是存一个链表的头节点。直接把新元素挂在对应位置的链表后面。**

==**B. 开放定址法—— 内存友好的做法**==

**所有元素都存在数组里，不使用链表。发生冲突后按照某种规则去找下一个空位置。线性探测：下标 +1, +2, +3... 挨个往后找。二次探测：下标 $+1^2, -1^2, +2^2, -2^2...$ 这样跳着找。**

3️⃣==**装填因子**==

**装填因子 $\alpha$ 的定义是：==已存储的元素个数 / 哈希表的总容量==**

**在链地址法中，$\alpha$ 的含义是：==每个桶位下挂载链表的平均长度。==**

| **特性**       | **开放定址法**                                    | **链地址法**                                           |
| -------------- | ------------------------------------------------- | ------------------------------------------------------ |
| **理论极限**   | $\alpha$ 绝对不能超过 1.0（数组满了就没地儿放了） | $\alpha$ 可以大于 1.0（链表可以无限长）                |
| **性能崩溃点** | 当 $\alpha > 0.7$ 时，冲突剧增，性能急剧下降      | 当 $\alpha$ 过大时，性能平缓下降（退化为链表）         |
| **扩容阈值**   | 通常设在 **0.5 ~ 0.7** 之间                       | C++ STL 通常设在 **1.0**（即平均每个桶一个元素时扩容） |

**开放地址法**需要不断扩容，且多线程不安全(锁粒度大，全局表锁)，**链式哈希**(锁粒度小，给每个哈希桶加锁就行)



### 一致性哈希算法

**==一致性哈希算法是分布式负载均衡的核心算法，主要解决传统哈希取模算法在节点增减时导致的大规模数据迁移问题。==传统的哈希取模算法一旦服务器数量 $N$ 发生变化（增加或宕机），几乎所有数据的映射位置都会改变。**



```c++

5.节点上线：如果新增服务器 D，它只会“抢夺”顺时针方向前一段距离的数据。
*/
```

2️⃣==**位图**==

==**位图法，就是使用一个位(0或1)来存储数据的状态**==，适合状态简单，数据量大，内存使用率低的场景。

```c++
/*
**位图法解决问题，首先需要知道待处理数据中的最大值，然后按照 size = (maxNumber/8)+1 的大小来开辟一个char类型的数组。当需要查找某个元素在位图中是否存在时，首先计算该数字对应位图中的比特位，然后读取值，0表示不存在，1表示存在。数据稀疏时内存利用率低。**

查找或者插入元素时：
1.找到数组下标：N / 8
2.找到位偏移：N % 8
映射到元素的位，读取该位的值（位运算），0表示未出现过，1表示出现过
*/

int main() {
    vector<int> v{12, 78, 90, 78, 123, 8, 9, 90}; // 目标：找第一个重复的数字   
    // 1. 寻找最大值以确定位图范围
    int maxVal = v[0];
    for (int val : v) {
        if (val > maxVal) maxVal = val;
    }
    // 2. 开辟位图空间 (char 数组)
    // 注意：new char[size]() 括号会把数组初始化为 0
    int size = maxVal / 8 + 1;
    char* bitmap = new char[size]()
    // 3. 遍历并查重
    for (int val : v) {
        int index = val / 8;     // 对应 char 数组下标
        int offset = val % 8;    // 对应字节内的比特偏移
        // 判断该位是否为 1
        if ((bitmap[index] & (1 << offset)) == 0) {
            // 如果是 0，说明没出现过，现在标记为 1 (用按位或运算)
            bitmap[index] |= (1 << offset);
        } else {
            // 如果是 1，说明之前已经存过了
            cout << val << " 是第一个重复的数字" << endl;
            break; // 找到第一个就跳出
        }
    }
    // 4. 释放内存
    delete[] bitmap;    
    return 0;
}
```

2️⃣**==快排分割==**

**利用快排分割函数每次返回基准数的位置，找出前 Top k 大的或者前 Top k 小的数据。**

```c++
/*
快排的 partition 函数每次执行完，都会选定一个基准数（pivot），并将其放在它最终排序后应该在的位置 pos。
如果 pos == k：恭喜，pos 左边的所有数就是最小的 k 个数。
如果 pos > k：说明第 k 个数在左半部分，我们递归去左边找。
如果 pos < k：说明第 k 个数在右半部分，我们递归去右边找。
*/

int partition(vector<int>& nums, int left, int right) {
    int pivot = nums[right];
    int i = left - 1;

    for (int j = left; j < right; j++) {
        // 如果发现当前数比基准数小
        if (nums[j] < pivot) {
            i++; // 先把“小于区”的边界扩充一格
            swap(nums[i], nums[j]); // 把这个小值换到扩充出来的格子里
        }
    }
    swap(nums[i + 1], nums[right]);
    return i + 1;
}
void quickSelect(vector<int>& nums, int left, int right, int k) {
    if (left >= right) return;
    int pos = partition(nums, left, right);
    if (pos == k) {
        return; // 基准数正好卡在第 K 个位置，结束
    } else if (pos > k) {
        quickSelect(nums, left, pos - 1, k); // 去左边找
    } else {
        quickSelect(nums, pos + 1, right, k); // 去右边找
    }
}
int main() {
    vector<int> vec{3, 2, 1, 5, 6, 4};
    int k = 3; // 找出最小的 3 个数
    quickSelect(vec, 0, vec.size() - 1, k - 1);
    for (int i = 0; i < k; ++i) {
        cout << vec[i] << " ";
    }
    return 0;
}
```



### 二叉树和平衡树

#### ==BST二叉搜索树==

**==二叉搜索树（BST）==：节点的值大于左孩子的值，小于右孩子的值。**

```c++
/*
BST树的插入：
1.根节点为空：生成新的节点
2.根节点不为空：从根节点开始比较，找到合适的位置，生成新的节点并插入
BST树的删除：
1.没有孩子的节点：把父节点置为nullptr
2.有一个孩子的节点：孩子写入父节点地址域
3.有两个孩子的节点：用前驱节点(左子树值最大节点)或后继节点(右子树值最小节点)的值把待删除节点的值覆盖，然后直接删除前驱节点或者后继节点--前驱或后继节点至多有一个孩子
*/
------------------------------------------------------------------------------------
// 1.BST树插入🌟🌟🌟🌟🌟
Node* insert(Node* root, int val) {
    // 1. 找到空位了，这就是新节点该待的地方
    if (root == nullptr) {
        return new Node(val);
    }
    // 2. 递归查找插入位置
    if (val < root->val) {
        // 如果比当前节点小，往左走
        // 关键点：把递归返回的新子树根节点，接回到当前节点的左指针上
        root->left = insert(root->left, val);
    } else if (val > root->val) {
        // 如果比当前节点大，往右走
        root->right = insert(root->right, val);
    }   
    // 3. 如果 val == root->val，通常 BST 不插入重复值，直接返回 root
    return root; 
}
------------------------------------------------------------------------------------
// 2.BST树删除🌟🌟🌟🌟🌟
Node* remove(Node* root, int key) {
    if (root == nullptr) return nullptr;
    // --- 第一部分：前序找节点 ---
    if (key < root->val) {
        root->left = remove(root->left, key); // 去左子树删，并把删完后的新左子树接回来
    } else if (key > root->val) {
        root->right = remove(root->right, key); // 去右子树删，并把删完后的新右子树接回来
    } 
    // --- 第二部分：找到目标，开始处理 (后序挂载逻辑) ---
    else {
        // 情况 1 & 2：无孩子或只有一个孩子
        if (root->left == nullptr) {
            Node* temp = root->right;
            delete root;
            return temp; //返回给父节点，父节点会把它接在 root->left 或 root->right 上
        } else if (root->right == nullptr) {
            Node* temp = root->left;
            delete root;
            return temp;
        }
        // 情况 3：有两个孩子
        // 找到右子树的最小节点（后继节点）
        Node* temp = findMin(root->right);
        // 用后继节点的值覆盖当前节点
        root->val = temp->val;
        // 转化为：在右子树中删除那个重复的后继节点
        root->right = remove(root->right, temp->val);
    }
    return root;
}
// 辅助函数：找子树的最小值
Node* findMin(Node* node) {
    while (node->left != nullptr) node = node->left;
    return node;
}

void levelOrder(Node* root,vector<int>& res){
    if(root == nullptr){
        return;
    }
    queue<Node*> que;
    que.push(root);
    
    while(!que.empty()){
        int len = que.size();
        for(int i = 0;i < len;i++){
            Node* cur = que.front();
            que.pop();
            res.push_back(cur->val);
            if(cur->left) que.push(cur->left);
            if(cur->right) que.push(cur->right);
        }
    }
}

// 5.前序遍历-->非递归实现🌟🌟🌟🌟🌟
void preOrder(Node* root,vector<int>& res){
    if(root == nullptr){
        return;
    }
    stack<Node*> st;
    st.push(root);   
    while(!st.empty()){
        // 1. 弹出栈顶元素，即为当前子树的“根”
        Node* cur = st.top();
        st.pop();
        // 2. 访问根节点
        res.push_back(cur->val);
        // 3. 关键点：先压入右孩子 (如果存在)
        // 因为栈是后进先出，右孩子先进去，就会被压在下面
        if(cur->right){
            st.push(cur->right);
        }
        if(cur->left){
            st.push(cur->left);
        }   
    }
}
// 5.中序遍历-->非递归实现🌟🌟🌟🌟🌟
void inOrder(Node* root, vector<int>& res) {
    stack<Node*> st;
    Node* cur = root;   
    while (cur != nullptr || !st.empty()) {
        // 步骤 1：尽可能往左走，沿途把节点都压入栈
        while (cur != nullptr) {
            st.push(cur);
            cur = cur->left;
        }    
        // 步骤 2：左边到头了，弹出栈顶（即当前最左的节点）
        cur = st.top();
        st.pop();
        res.push_back(cur->val); // 访问 
        // 步骤 3：转向右子树，开始新一轮的“向左钻”
        cur = cur->right;
    }
}
// 5.后序遍历-->非递归实现🌟🌟🌟🌟🌟
void postorderTraversal(Node* root,vector<int>& res) {
    if (!root) return;
    stack<Node*> st;
    st.push(root);
    while (!st.empty()) {
        Node* cur = st.top();
        st.pop();
        res.push_back(cur->val); // 根
        if (cur->left) st.push(cur->left);   // 左先入栈，右就会先出栈
        if (cur->right) st.push(cur->right); // 右后入栈，下一轮先处理右
    }
    // 此时 res 是 根->右->左
    reverse(res.begin(), res.end()); // 翻转后变成 左->右->根
}

void findRangeValue(Node* root, vector<int>& res, int i, int j) {//剪枝操作
    if (root == nullptr) return;
    // 只有当当前节点值 > 最小值 i 时，左子树才可能存在区间内的值
    if (root->val > i) {
        findRangeValue(root->left, res, i, j);
    }
    // 处理当前节点
    if (root->val >= i && root->val <= j) {
        res.push_back(root->val);
    }
    // 只有当当前节点值 < 最大值 j 时，右子树才可能存在区间内的值
    if (root->val < j) {
        findRangeValue(root->right, res, i, j);
    }
}

// 7.判断二叉树是否是一颗BST树🌟🌟🌟🌟🌟
long long lastVal = LLONG_MIN; // 用一个最小值记录前驱节点的值
bool isValidBST(Node* root) { //当前节点的值一定要大于上一个节点的值
    if (root == nullptr) return true;
    // 1. 访问左子树
    if (!isValidBST(root->left)) return false;
    // 2. 比较当前节点：必须大于上一个节点值
    if (root->val <= lastVal) return false;
    lastVal = root->val; // 更新前驱节点值
    // 3. 访问右子树
    return isValidBST(root->right);
}

// 8.判断二叉树是不是另一颗二叉树的子树🌟🌟🌟🌟🌟
bool isSubtree(Node* s, Node* t) {
    // 规定：空树是任何树的子树（取决于题目要求，通常这样处理）
    if (t == nullptr) return true;
    // s 已经走完了还没匹配上，返回 false
    if (s == nullptr) return false;
    // 1. 如果当前节点值相等，尝试以此为起点进行匹配
    if (s->val == t->val) {
        if (isSameTree(s, t)) return true;
    }
    // 2. 如果当前节点不匹配，递归地去 s 的左子树和右子树里找
    return isSubtree(s->left, t) || isSubtree(s->right, t);
}
// 辅助函数：判断两棵树是否完全相等
bool isSameTree(Node* s, Node* t) {
    // 两个都为空，说明匹配成功
    if (s == nullptr && t == nullptr) return true;
    // 其中一个为空，或者值不等，说明不匹配
    if (s == nullptr || t == nullptr || s->val != t->val) return false;    
    // 递归比较左右子树
    return isSameTree(s->left, t->left) && isSameTree(s->right, t->right);
}

// 9.二叉树的最近公共祖先🌟🌟🌟🌟🌟
Node* lowestCommonAncestor(Node* root, Node* p, Node* q) {
    // 1. 边界条件：如果撞到了空节点，或者撞到了 p 或 q，直接返回当前节点
    if (root == nullptr || root == p || root == q) {
        return root;
    }
    // 2. 递归去左子树和右子树里找
    Node* left = lowestCommonAncestor(root->left, p, q);
    Node* right = lowestCommonAncestor(root->right, p, q);
    // 3. 后序逻辑：根据左右子树的返回值来判断（回溯）    
    // 情况 A: 左右子树都找到了东西，说明 p 和 q 分布在当前节点两侧
    // 那么当前节点就是 LCA！
    if (left != nullptr && right != nullptr) {
        return root;
    }
    // 情况 B: 只有左子树找到了，说明 p 和 q 都在左边，向上返回找到的那个
    if (left != nullptr) {
        return left;
    }
    // 情况 C: 只有右子树找到了，向上返回那个
    return right;
}

// 11.BST树的镜像对称问题(根节点的左右子树是否是对称的)🌟🌟🌟🌟🌟
bool isSymmetric(Node* root) {
    if (root == nullptr) return true;
    return isMirror(root->left, root->right);
}
bool isMirror(Node* p, Node* q) {
    // 1. 同时为空，对称
    if (p == nullptr && q == nullptr) return true;
    // 2. 一个为空一个不为，或者值不等，不对称
    if (p == nullptr || q == nullptr || p->val != q->val) return false;
    // 3. 关键：同步递归比较
    // p 的左边 vs q 的右边
    // p 的右边 vs q 的左边
    return isMirror(p->left, q->right) && isMirror(p->right, q->left);
}

// 12.前序遍历与中序遍历重建二叉树🌟🌟🌟🌟🌟
Node* build(vector<int>& pre,vector<int>& in){
    if(pre.size() == 0){
        return nullptr;
    }
    int val = pre[0];
    int index = 0;
    for(int i = 0;i < in.size();i++){
        if(val == in[i]){
            index = i;
        }
    }
    vector<int> preleft(pre.begin()+1,pre.begin()+index+1);
    vector<int> preright(pre.begin()+index+1,pre.end());
    vector<int> inleft(in.begin(),in.begin()+index);
    vector<int> inright(in.begin()+index+1,in.end());
    Node* root = new Node(val);
    root->left = build(preleft,inleft);
    root->right = build(preright,inright);
    return root;
}

// 13.判断二叉树是否是平衡树(任意节点的子树高度差不能超过1)🌟🌟🌟🌟🌟

int getHeight(Node* root) { // 辅助函数：计算高度，如果不平衡则返回 -1
    if (root == nullptr) return 0;
    // 1. 递归获取左子树高度
    int leftHeight = getHeight(root->left);
    if (leftHeight == -1) return -1; // 左子树已经不平衡了，直接透传 -1
    // 2. 递归获取右子树高度
    int rightHeight = getHeight(root->right);
    if (rightHeight == -1) return -1; // 右子树已经不平衡了，直接透传 -1
    // 3. 后序逻辑：判断当前节点是否平衡
    // 左右高度差绝对值 > 1 则不平衡
    if (abs(leftHeight - rightHeight) > 1) {
        return -1;
    }
    // 4. 如果平衡，返回当前节点的高度
    return max(leftHeight, rightHeight) + 1;
}

Node* kthLargest(Node* root, int k) {//中序遍历下的倒数第 k个其实就是右-根-左顺序下的第 k个
    stack<Node*> st;
    Node* cur = root;
    int count = 0;
    while (cur || !st.empty()) {
        // 1. 一路向右钻到底
        while (cur) {
            st.push(cur);
            cur = cur->right;
        }
        // 2. 弹出并处理
        cur = st.top();
        st.pop();        
        count++;
        if (count == k) return cur;
        // 3. 转向左子树
        cur = cur->left;
    }
    return nullptr;
}
```

#### ==AVL二叉平衡树==

**AVL 树是一个 BST 树，且任意节点的左子树与右子树的高度差不超过1。**

**当你在 AVL 树中插入或删除节点，导致某个节点的平衡因子变成 2 或 -2 时，它就会通过“旋转”来找回平衡**。旋转分为四种情况：

| **情况**  | **现象**           | **解决方法**       |
| --------- | ------------------ | ------------------ |
| **LL 型** | 左孩子的左子树太高 | **右旋**           |
| **RR 型** | 右孩子的右子树太高 | **左旋**           |
| **LR 型** | 左孩子的右子树太高 | **先左旋，再右旋** |
| **RL 型** | 右孩子的左子树太高 | **先右旋，再左旋** |

**==通过失衡节点（根节点）和导致失衡的那个子节点的平衡因子来判断属于哪种失衡类型==**。

#### ==红黑树==

1️⃣**==每一个节点都有颜色：==不是红色，就是黑色。**

2️⃣==**根节点必为黑色**：==**树的起始点是稳固的黑色**。

3️⃣==**叶子节点（NULL）必为黑色**：==**这里的叶子是指空节点**。

4️⃣==**红色节点不连续**：==**红色节点的两个子节点必须是黑色（即：不能有两个红节点连在一起）。**

5️⃣==**黑高一致**：==**从任一节点到其每个叶子的所有路径，包含相同数目的黑色节点。**

| **维度**         | **AVL 树**                    | **红黑树 (RBT)**                              |
| ---------------- | ----------------------------- | --------------------------------------------- |
| **平衡程度**     | **严格平衡** (高度差 $\le 1$) | **弱平衡** (最长路径 $\le 2 \times$ 最短路径) |
| **增删查复杂度** | **$O(\log n)$**               | **$O(\log n)$**                               |
| **插入旋转次数** | **最多 2 次(需要先查找)**     | **最多 2 次(需要先查找)**                     |
| **删除旋转次数** | **最多 $O(\log n)$ 次**       | **最多 3 次(需要先查找)**                     |

**AVL 查找更快，红黑树对插入/删除更友好**

#### ==回溯算法==

**算法思想：**在包含问题的所有解的解空间树中，按照深度优先搜索的策略，从根节点出发深度搜索解空间树。当搜索到某一节点时，要先判断该节点是否包含问题的解，如果包含就从该节点出发继续深度搜索下去，否则逐层向上回溯。一般在搜索的过程中都会添加相应的剪枝函数，避免无效解的搜索，提高算法效率。

**解空间：**解空间就是所有解的可能取值构成的空间，一个解往往包含了得到这个解的每一步，往往就是对应解空间树中一条从根节点到叶子节点的路径。子集树和排列树都是一种解空间，它们不是真实存在的数据结构，也就是说并不是真的有这样一棵树，只是抽象出的解空间树。

```c++
// 🌟🌟🌟🌟🌟1.整数选择问题：给定一组整数，从里面挑选出一组整数，让选择的整数的和，与剩下的整数的和的差最小
//将其转化为：寻找一个子集，其总和尽可能接近整个数组总和的一半 (Sum / 2)。
long long minDiff = LLONG_MAX;
vector<int> bestSubset;  // 存储最优的一组解
long long totalSum = 0;
void backtrack(vector<int>& arr, int start, long long curSum, vector<int>& path) {
    // 1. 在线评估：计算当前组合与剩余部分的差值
    long long diff = abs(totalSum - 2 * curSum);    
    // 如果发现了更小的差值，立即更新“最佳结果”
    if (diff < minDiff) {
        minDiff = diff;
        bestSubset = path; // 记录当前这个最优子集
    }
    // 2. 终止条件：如果已经没有数字可选了，返回
    if (start == arr.size()) {
        return;
    }
    // 3. 遍历选择列表
    for (int i = start; i < arr.size(); i++) {
        // --- 前序位置：做选择 ---
        path.push_back(arr[i]);
        curSum += arr[i];
        // 递归进入下一层
        backtrack(arr, i + 1, curSum, path);
        // --- 后序位置：撤销选择 (回溯) ---
        curSum -= arr[i];
        path.pop_back();
    }
}

long long totalSum = 0;
void backtrack(vector<int>& arr, int start, int count, long long curSum, vector<int>& path, int n) {
    // 1. 显式终止条件：已经选够了 n 个
    if (count == n) {
        long long diff = abs(totalSum - 2 * curSum);
        if (diff < minDiff) {
            minDiff = diff;
            bestSubset = path;
        }
        return;
    }
    // 2. 剪枝优化：如果后面所有的数字都加上，也不够 n 个，直接返回
    // 剩余可选个数：arr.size() - start
    // 还需要个数：n - count

        return;

        path.push_back(arr[i]);

        path.pop_back();
    }
}

void backtrack(const vector<int>& arr, int start, int target, int curSum, vector<int>& path, vector<vector<int>>& res) {
    // 1. 达成目标：记录结果
    if (curSum == target) {
        res.push_back(path);
        return; // 找到了一个，返回去试别的路径
    }
    // 2. 剪枝：如果已经超了或者没数了 (前提：arr中都是正数)
    if (curSum > target || start == arr.size()) {
        return;
    }
    // 3. 标准回溯：地毯式搜索
    for (int i = start; i < arr.size(); i++) {
        // 做选择
        path.push_back(arr[i]);
        // 递归：寻找包含当前数字的所有可能解
        backtrack(arr, i + 1, target, curSum + arr[i], path, res);
        // 撤销选择：回溯，腾出位置给下一个数
        path.pop_back();
    }
}

void backtrack(const vector<int>& w, const vector<int>& v, int start, int curW, int curV, int c) {

    if (curV > bestValue) {

    // 2. 终止条件（写不写均可，因为 for 循环有自然边界，但写了更清晰）
    if (start == w.size()) {

    // 3. 标准模板：从当前位置向后尝试每一个物品
    for (int i = start; i < w.size(); i++) {
        // --- 剪枝：如果装不下这个，就跳过（注意：i+1 还要继续试别的） ---
        if (curW + w[i] <= c) {
            // 【前序位置】：做选择
            curW += w[i];
            curV += v[i];
            // 【递归】：继续挑下一个（从 i + 1 开始）
            backtrack(w, v, i + 1, curW, curV, c);
            // 【后序位置】：撤销选择（回溯）
            curW -= w[i];
            curV -= v[i];
        }
    }
}

// 全排列：[1, 2]和 [2, 1]是两个不同的排列（所以我们每一层都要从头开始选，只要这个数还没被用过）。
void backtrack(vector<int>& nums, vector<bool>& used, vector<int>& path, vector<vector<int>>& res) {
    // 1. 终止条件：路径长度等于数组长度，说明选够了
    if (path.size() == nums.size()) {
        res.push_back(path);
        return;
    }
    // 2. 遍历选择列表：注意！这里每次都从 0 开始
    for (int i = 0; i < nums.size(); i++) {
        // --- 核心约束：如果这个数已经被选过了，就跳过 ---
        if (used[i]) continue;
        // 做选择
        used[i] = true;
        path.push_back(nums[i]);
        // 递归：进入下一层决策
        backtrack(nums, used, path, res);
        // 撤销选择（回溯）
        path.pop_back();
        used[i] = false;
    }
}
```

#### ==分治算法==

**分治算法思想：**规模为n的原问题的解无法直接求出，进行问题规模缩减，划分子问题（这里子问题相互独立而且和原问题解的性质是相同的，只是问题规模缩小了）。如果子问题的规模仍然不够小，再进行子问题划分，如此递归的进行下去，直到问题的解足够小，很容易求出其解为止，最后将求出的小规模的问题的解合并为一个更大规模的问题的解，自底向上逐步求出原问题的解。

```c++
-----------------------------------------------------------------------------------
// 🌟🌟🌟🌟🌟快速排序
int partition(vector<int>& arr, int low, int high) {
    // 简单起见，取最后一个元素作为基准
    int pivot = arr[high]; 
    int i = low - 1; // 🌟🌟🌟🌟🌟 i 是小于基准区域的右边界
    for (int j = low; j < high; j++) {
        // 如果当前元素小于或等于基准
        if (arr[j] <= pivot) {
            i++;
            swap(arr[i], arr[j]);
        }
    }
    // 最后把基准元素交换到中间
    swap(arr[i + 1], arr[high]);
    return i + 1; // 返回基准的最终位置
}
void quickSort(vector<int>& arr, int low, int high) {
    if (low < high) {
        // 获取分区点
        int pi = partition(arr, low, high);
        // 递归排序左半部分和右半部分
        quickSort(arr, low, pi - 1);
        quickSort(arr, pi + 1, high);
    }
}

// 🌟🌟🌟🌟🌟归并排序
// 合并两个有序区间：[left, mid] 和 [mid + 1, right]
void merge(vector<int>& arr, int left, int mid, int right) {
    vector<int> temp;    // 创建临时容器存放合并结果
    int i = left;      // 左序列指针
    int j = mid + 1;   // 右序列指针
    // 比较两个序列，谁小就谁先入场
    while (i <= mid && j <= right) {
        if (arr[i] <= arr[j]) {
            temp.push_back(arr[i++]);
        } else {
            temp.push_back(arr[j++]);
        }
    }
    // 处理剩余元素
    while (i <= mid) temp.push_back(arr[i++]);
    while (j <= right) temp.push_back(arr[j++]);
    // 将临时容器里的有序数据写回原数组
    for (int k = 0; k < temp.size(); ++k) {
        arr[left + k] = temp[k];
    }
}

void mergeSort(vector<int>& arr, int left, int right) {
    if (left >= right) return;

    int mid = left + (right - left) / 2; // 防止溢出

    mergeSort(arr, left, mid);      // 递归左半部分
    mergeSort(arr, mid + 1, right); // 递归右半部分
    
    merge(arr, left, mid, right);   // 合并左右两部分
}
```

#### ==动态规划算法==

**算法思想：**将待求解的问题划分为若干子问题，按划分的顺序求解子阶段问题，前一个子问题的解，为后一个子问题的求解提供了有用的信息。

```c++
/*
1️⃣定义状态
你需要用一个变量（通常是数组 dp[i] 或 dp[i][j]）来表示：当我走到这一步时，我手里掌握的最优解是什么？
--例子（0-1 背包）：dp[i][w] 表示前 i 件物品在限重为 w 时的最大价值。
--例子（打家劫舍）：dp[i] 表示偷到第 i 家时能拿到的最高金额。
2️⃣推导状态转移方程
这是 DP 的灵魂。你要思考：“我现在的这个状态，是怎么由之前的状态推导出来的？”
它本质上是一个逻辑选择：
dp[i] = max( 方案 A, 方案 B )
比如背包问题：要么我不拿第 i 个物品（继承 dp[i-1]），要么我拿（v[i] + 之前的剩余空间价值）。
3️⃣确定初始状态
这是你的起点，也就是最简单的情况。
比如 dp[0] = 0（没物品时价值为0）。
4️⃣确定遍历顺序
你需要决定是从前往后填表，还是从后往前，或者是斜着填。
原则：当你计算 dp[i] 时，它依赖的那些状态（比如 dp[i-1]）必须已经计算好了。
5️⃣举例推导
在动手写代码前，先在纸上画个表格，手动填几个数。如果你手填的结果和你的方程对不上，说明你的状态定义或者转移逻辑出错了。
*/

// 🌟🌟🌟🌟🌟硬币选择问题：有 1,3,5 分面额的硬币，给定一个面值11，问组成给定面值所需要的最少的硬币数量是多少
int coinChange(int amount) {
    // 1. 初始化 dp 数组，面值为 i 的最少硬币数
    // 初始化为 amount + 1，相当于无穷大
    vector<int> dp(amount + 1, amount + 1);    
    // 2. 初始状态
    dp[0] = 0;
    // 3. 遍历顺序：从小到大凑面值
    for (int i = 1; i <= amount; i++) {
        // 尝试每一种硬币
        if (i >= 1) dp[i] = min(dp[i], dp[i - 1] + 1);
        if (i >= 3) dp[i] = min(dp[i], dp[i - 3] + 1);
        if (i >= 5) dp[i] = min(dp[i], dp[i - 5] + 1);
    }
    return dp[amount] > amount ? -1 : dp[amount];
}

// 🌟🌟🌟🌟🌟斐波那契数列：f(0)=1,f(1)=1,...,f(n)=f(n-1)+f(n-2),求f(20)

// 🌟🌟🌟🌟🌟最大子段和：给定n个整数（可能为负数），求子段和的最大值，当所给的子段均为负数时定义子段和为0
int maxSubArray(vector<int>& nums) {
    int n = nums.size();
    if (n == 0) return 0;
    // 1. 定义状态数组：dp[i] 表示以 nums[i] 结尾的最大子段和
    vector<int> dp(n);
    // 2. 初始状态
    dp[0] = nums[0];
    int res = dp[0]; // res 用来记录 dp 数组里的最大值
    // 3. 填表
    for (int i = 1; i < n; i++) {
        // 状态转移方程：
        // 如果 dp[i-1] > 0，说明对我有帮助，我接着；
        // 如果 dp[i-1] <= 0，说明是累赘，我直接从 nums[i] 重新开始。
        dp[i] = max(nums[i], dp[i - 1] + nums[i]);       
        // 记录全局最大值
        res = max(res, dp[i]);
    }
    // 题目要求：如果全为负数，定义为 0
    return res < 0 ? 0 : res;
}

// 🌟🌟🌟🌟🌟最长非降（升序）子序列
int LIS(vector<int>& arr){
    int n = arr.size();
    if(n <= 1) return n;    
    int res = 1;
    vector<int> dp(n, 1); // 默认初始化为 1，这样就不需要 max+1 的逻辑了

    for(int i = 1; i < n; i++){
        for(int j = 0; j < i; j++){

                // 直接在 dp[i] 上更新最大值
                dp[i] = max(dp[i], dp[j] + 1);
            }
        }
        res = max(res, dp[i]);
    }
    return res;
}

// 🌟🌟🌟🌟🌟0-1背包问题：有一组物品其重量为w1,w2,...,wn,其价值为v1,v1,...,vn，现有一背包容量为c，怎样装入物品使得背包价值最大化？
int knapsack(int c, vector<int>& w, vector<int>& v) {
    int n = w.size();
    // 1. 初始化 dp 表：n+1 行（物品），c+1 列（重量）
    // dp[i][j] 默认都是 0
    vector<vector<int>> dp(n + 1, vector<int>(c + 1, 0));
    // 2. 开始填表
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= c; j++) {
            // 当前物品的重量和价值（注意索引偏移）
            int weight = w[i - 1];
            int value = v[i - 1];
            if (j < weight) {
                // 装不下，继承上方的状态
                dp[i][j] = dp[i - 1][j];
            } else {
                // 装得下，在“放”与“不放”之间选大的
                dp[i][j] = max(dp[i - 1][j], value + dp[i - 1][j - weight]);
            }
        }
    }    
    // 右下角的格子就是最终答案
    return dp[n][c];
}   
void findItems(int c, vector<int>& w, vector<int>& v, vector<vector<int>>& dp) {//最终选择的物品
    int n = w.size();
    int j = c; // 当前剩余容量
    vector<int> selected;
    // 从最后一个物品开始往前找
    for (int i = n; i >= 1; i--) {
        // 如果当前格子的值不等于上一行的值，说明装了第 i 个物品
        if (dp[i][j] != dp[i - 1][j]) {
            selected.push_back(i - 1); // 记录物品索引
            j -= w[i - 1];             // 减去重量，继续找剩下的
        }
    }
    // 输出结果
    cout << "放入背包的物品索引为: ";
    for (int idx : selected) {
        cout << idx << " ";
    }
    cout << endl;
}

// 🌟🌟🌟🌟🌟给定一个三角形，找出自顶向下的最小路径和，每一步只能移动到下一行中相邻的节点上
int minimumTotal(vector<vector<int>>& triangle) {
    int n = triangle.size();
    // 创建一个和 triangle 形状一样的 dp 数组
    vector<vector<int>> dp(n, vector<int>(n));//🌟🌟🌟    
    dp[0][0] = triangle[0][0];
    for (int i = 1; i < n; i++) {
        for (int j = 0; j <= i; j++) {
            if (j == 0) {
                // 最左边，只能从正上方下来
                dp[i][j] = dp[i - 1][j] + triangle[i][j];
            } else if (j == i) {
                // 最右边，只能从左上方下来
                dp[i][j] = dp[i - 1][j - 1] + triangle[i][j];
            } else {
                // 中间部分，挑一个小的
                dp[i][j] = min(dp[i - 1][j - 1], dp[i - 1][j]) + triangle[i][j];
            }
        }
    }
    // 最后，在最后一行找一个最小值
    int res = dp[n - 1][0];
    for (int j = 1; j < n; j++) {
        res = min(res, dp[n - 1][j]);
    }
    return res;
}
```

```c++

// 🌟🌟🌟🌟🌟部分背包问题：有n个物品，第i个物品的重量为wi，价值为vi,在总重量不超过c的情况下让总价值尽量高。每个物品都可以只取走一部分，价值和重量按比例计算
struct Item {
    int id;
    double weight;
    double value;
    double ratio; // 单位价值（性价比）
};
double fractionalKnapsack(int c, vector<int>& w, vector<int>& v) {
    int n = w.size();
    vector<Item> items(n);    
    for (int i = 0; i < n; i++) {
        items[i] = {i, (double)w[i], (double)v[i], (double)v[i] / w[i]};
    }
    // 1. 贪心策略：按性价比排序
    sort(items.begin(), items.end(), [](const Item& a,const Item& b){return a.ratio > b.ratio;});
    double totalValue = 0.0;
    double currentWeight = 0.0;
    for (int i = 0; i < n; i++) {
        if (currentWeight + items[i].weight <= c) {
            // 2. 如果能全部装下，就全装
            currentWeight += items[i].weight;
            totalValue += items[i].value;
        } else {
            // 3. 如果装不下了，切碎它填满剩余空间
            double remaining = c - currentWeight;
            totalValue += items[i].ratio * remaining;
            break; // 背包满了
        }
    }
    return totalValue;
}
```

#### ==KMP算法==

```c++
/*
KMP 算法的核心思想是：主串指针不回退，利用已经匹配过的信息，让模式串尽可能多地向右“滑动”。
当我们在模式串的第 j 位失配时，KMP 不会把模式串移回起点，而是去问：“在刚才匹配成功的这一段里，最长的一模一样的前(包含第一个字符，但不包含最后一个字符的子串)和后缀(包含最后一个字符，但不包含第一个字符的子串)是多少？”
next数组：当匹配失败时，告诉子串应该跳到哪里重新开始匹配，next[i] 代表的是：子串前 i+1 个字符中，最长的“相等前后缀”的长度。

匹配流程：
预处理：计算模式串的 next 数组。
匹配过程：当 S[i] != P[j] 时：主串 i 不动。模式串 j 跳到 next[j-1] 的位置继续比。
*/

void getNext(vector<int>& next, const string& s) {
    int n = s.size();
    int j = 0;   // 🌟🌟j: 前缀末尾指针，同时也代表当前最长相等前后缀的长度
    next[0] = 0; // 单个字符没有前后缀，长度必为 0
    for (int i = 1; i < n; i++) { // 🌟🌟i: 后缀末尾指针
        // 【情况一】前后缀不匹配：j 连续回退
        // 就像在“前缀仓库”里找备胎，直到找到匹配或者 j 退回起点
        while (j > 0 && s[i] != s[j]) {
            j = next[j - 1]; 
        }
        // 【情况二】前后缀匹配：j 长度增加
        if (s[i] == s[j]) {
            j++;
        }
        // 【赋值】记录当前 i 位置下的最长相等前后缀长度
        next[i] = j;
    }
}

int strStr(string s, string p) {
    if (p.empty()) return 0;
    int n = s.size();
    int m = p.size();
    // 1. 预处理模式串，得到 next 数组
    vector<int> next(m);
    getNext(next, p);
    
    int j = 0; // 模式串的指针
    for (int i = 0; i < n; i++) { // i 是主串指针，永不回退        
        // 如果当前字符不匹配，模式串指针 j 根据 next 数组跳跃
        while (j > 0 && s[i] != p[j]) {
            j = next[j - 1];
        }
        // 如果字符匹配，模式串指针 j 前进一位
        if (s[i] == p[j]) {
            j++;
        }
        // 如果 j 到达模式串末尾，说明匹配成功
        if (j == m) {
            return i - m + 1; // 返回起始位置
        }
    }
    return -1;
}
```

### 跳跃表 SkipList

```c++
/*
跳跃表 SkipList：通过给有序链表增加“多层索引”，实现类似折半查找的效率。
L3 (索引层2):  1 ---------------------------> 15 ----------------> NULL
                |                             |
L2 (索引层1):  1 -----------> 7 ------------> 15 -------> 19 ----> NULL
                |             |               |          |
L1 (原始链表):  1 -> 3 -> 7 -> 8 -> 9 -> 12 -> 15 -> 18 -> 19 -> 21 -> NULL
🌟🌟查找 19 的路径：
1️⃣从 L3 开始：在 1 处，看下一个是 15。19 > 15，跳到 15。
2️⃣在 L3 15 处，看下一个是 NULL。向下切到 L2。
3️⃣在 L2 15 处，看下一个是 19。正好匹配！ 结果： 只用了很短的步数就定位到了目标。
🌟🌟节点的插入：
1️⃣先在最底层插入。
2️⃣抛硬币：如果是正面，节点就“长高”一层，建立一个索引；如果是反面，停止。
3️⃣重复抛硬币，直到出现反面。

*/
const int MAX_LEVEL = 16;    // 最大层数限制
const float P = 0.5;         // 晋升概率（硬币正面概率）

struct Node {
    int value;
    // forward[i] 表示该节点在第 i 层的下一个节点
    vector<Node*> forward;
    Node(int val, int level) : value(val), forward(level + 1, nullptr) {}
};

class SkipList {
private:
    Node* head;
    int currentLevel; // 当前跳表最高的层数
    // 核心：抛硬币逻辑
    int randomLevel() {
        int lvl = 0;
        // 只要随机数小于概率P且没超过最大限制，就一直增加高度
        while ((float)rand() / RAND_MAX < P && lvl < MAX_LEVEL - 1) {
            lvl++;
        }
        return lvl;
    }
public:
    SkipList() {
        head = new Node(-1, MAX_LEVEL); // 头节点不存值
        currentLevel = 0;
        srand((unsigned)time(NULL));
    }
    // 查找
    bool search(int target) {
        Node* curr = head;
        // 从最高层向下找
        for (int i = currentLevel; i >= 0; i--) {
            while (curr->forward[i] && curr->forward[i]->value < target) {
                curr = curr->forward[i];
            }
        }
        curr = curr->forward[0];
        return curr && curr->value == target;
    }
    // 插入
    void insert(int val) {
        // update数组用来暂存每一层中“刚好小于val”的那个节点
        vector<Node*> update(MAX_LEVEL, nullptr);
        Node* curr = head;
        for (int i = currentLevel; i >= 0; i--) {
            while (curr->forward[i] && curr->forward[i]->value < val) {
                curr = curr->forward[i];
            }
            update[i] = curr;
        }
        int lvl = randomLevel();
        // 如果随机出的高度超过了当前最高高度，更新头节点指向
        if (lvl > currentLevel) {
            for (int i = currentLevel + 1; i <= lvl; i++) {
                update[i] = head;
            }
            currentLevel = lvl;
        }
        Node* newNode = new Node(val, lvl);
        for (int i = 0; i <= lvl; i++) {
            // 经典的链表插入：newNode->next = prev->next; prev->next = newNode;
            newNode->forward[i] = update[i]->forward[i];
            update[i]->forward[i] = newNode;
        }
    }
};
/*
1. 把它看作“超级快车”
想象你在坐地铁：
底层 (Level 0)：是“全站停”的慢车，每一站都停。
高层 (Level 1, 2...)：是“大站快车”或“直达特快”，只停大站。
查找过程：就是你先坐快车，发现快坐过头了，就赶紧下车，换乘下面一层的慢车继续找。

2. 记住那个“脚手架” (Update 数组)
代码里最绕的 update 数组，其实就是你在下楼梯时，在每一层拐角处踩下的“脚印”。
插入新节点时，你得回到这些脚印的位置，把新电梯的门一个个接上。

3. 不要纠结“为什么这层是 3 而不是 4”
跳表的精髓在于模糊的正确。 红黑树追求的是“绝对平衡”，所以它活得很累（代码极难写）。 跳表追求的是“大概平衡”，它相信概率。只要数据量够大，它自然就会变得高效。
```

### 哈夫曼树和哈夫曼编码

```c++
/*
哈夫曼树：最佳判定树，是一种带权路径长度最短的二叉树，常用于数据压缩。
带权路径长度：树中所有叶子节点的权值乘以其到根节点的路径长度
🌰例：假设我们要对以下字符频率进行编码： A: 5, B: 9, C: 12, D: 13, E: 16, F: 45（总权重 100）
构造过程：
--找两个权重最小的（A和B），合成一棵树，根节点权重 = 5+9=14
--将这个 14 放回去，继续找最小的两个
--重复直到剩下一棵树
			  (100)
             /     \
          0 /       \ 1
         (45:F)    (55)
                  /    \
               0 /      \ 1
              (25)      (30)
             /   \     /    \
          0 /   1 \ 0 /    1 \
         (12:C)(13:D)(14)   (16:E)
                    /   \
                 0 /     \ 1
                (5:A)    (9:B)
*/
// 构建哈夫曼树
struct Node {
    char ch;
    int freq;
    Node *left, *right;    
    Node(char c, int f) : ch(c), freq(f), left(nullptr), right(nullptr) {}
};
// 比较规则：频率小的先出队
struct compare {
    bool operator()(Node* l, Node* r) { return l->freq > r->freq; }
};
Node* buildHuffmanTree(priority_queue<Node*, vector<Node*>, compare>& minHeap) {
    while (minHeap.size() > 1) {
        // 取出最小的两个节点
        Node *left = minHeap.top(); minHeap.pop();
        Node *right = minHeap.top(); minHeap.pop();        
        // 创建新节点作为它们的父节点
        Node *parent = new Node('$', left->freq + right->freq);
        parent->left = left;
        parent->right = right;        
        minHeap.push(parent);
    }
    return minHeap.top(); // 最后剩下的就是根节点
}
```

```c++
/*🌰例：
```

```c++
// 不带权值最短路径----BFS---返回从 start 到 target 的最短路径列表
void printShortestPath(int start, int target, const vector<vector<int>>& adj) {
    int n = adj.size() - 1;
    vector<int> parent(n + 1, -1); // 记录每个节点的直接前驱
    vector<bool> visited(n + 1, false);
    queue<int> q;
    visited[start] = true;
    q.push(start);
    bool found = false;
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        if (u == target) {
            found = true;
            break; 
        }
        for (int v : adj[u]) {
            if (!visited[v]) {
                visited[v] = true;
                parent[v] = u; // 关键：记录 v 是从 u 过来的
                q.push(v);
            }
        }
    }
	// 如果没找到
    if (!found) {
        cout << "从 " << start << " 到 " << target << " 不存在路径" << endl;
        return;
    }
    // 找到了，通过 parent 数组回溯找路径
    vector<int> path;
    for (int v = target; v != -1; v = parent[v]) {
        path.push_back(v);
    }
    reverse(path.begin(), path.end()); // 因为是逆序回溯的，需要反转
    cout << "最短路径长度: " << path.size() - 1 << endl;
    cout << "路径轨迹: ";
    for (int i = 0; i < path.size(); ++i) {
        cout << path[i] << (i == path.size() - 1 ? "" : " -> ");
    }
    cout << endl;
}

/* 带权值(无负权边)的最短路径---迪杰斯特拉算法
假设起点是 S，我们要算它到其他点的距离。
1️⃣初始化距离数组 dist：记下起点到每个点的距离。起初，dist[S] = 0，其他所有点的距离都设为无穷大，访问标记 visited：准备一个集合，记录哪些点的“最短距离”已经板上钉钉了。前驱数组 parent（可选）：记录每个节点是从谁那里走过来的，方便最后还原路线。
2️⃣在所有还没标记为 visited 的节点中，找一个 dist 值最小的节点，把它记为 u。注意：第一次选出的肯定是起点 S 自己，因为距离是 0。
3️⃣把节点 u 标记为 visited。这个点的最短路径已经找到了，以后不用再动它了。
4️⃣遍历节点 u 的所有邻居 v。计算：从起点经过 u 到达 v 的总距离 = dist[u] + weight(u, v)。比比看：如果这个新距离比账本上原来记的 dist[v] 还要小，说明我们发现了一条“更省钱的绕路”。改账本：更新 dist[v] = dist[u] + weight(u, v)，并把 v 的爸爸记为 u。
5️⃣循环重复回到“第二步”，直到所有节点都变成了 visited。    
                     (1)       (4)         (20)
                [1] ----> [3] ----> [6] ----------> [10]
                 |         |         |               ^
                 | (10)    | (1)     | (15)          |
                 |         v         |               |
                 | ----> [2] <-------/               |
                 |         |                         |
                 | (5)     | (2)          (6)        |
                 v         v             /-----------/
                [4]       [5] --------> [7]*/
************************************************************
const int INF = 1e9; // 定义无穷大
// 执行 Dijkstra 算法
void dijkstra(int start, int target, int n, const vector<vector<pair<int, int>>>& adj) {
    // 账本：记录起点到每个点的最小路费
    vector<int> dist(n + 1, INF);
    // 记录路径：记录每个点是从谁那过来的
    vector<int> parent(n + 1, -1);    
    // 优先队列：{当前累计路费, 节点}
    // 使用 greater 会让队列从小到大排序，即“小顶堆”
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    // 起点初始化
    dist[start] = 0;
    pq.push({0, start});
    while (!pq.empty()) {
        int d = pq.top().first;
        int u = pq.top().second;
        pq.pop();
        // 核心优化：如果弹出的路费比账本里已经记的还贵，说明是过时的消息，扔掉
        if (d > dist[u]) continue;
        // 看看 u 的邻居们
        for (auto& edge : adj[u]) { // pair 存储 {目的地, 路费}
            int v = edge.first;
            int weight = edge.second;
            // 松弛操作：如果 走 u 到 v 发现更省钱了
            if (dist[u] + weight < dist[v]) {
                dist[v] = dist[u] + weight;
                parent[v] = u;
                pq.push({dist[v], v});
            }
        }
    }
    // --- 输出结果 ---
    if (dist[target] == INF) {
        cout << "无法到达节点 " << target << endl;
    } else {
        cout << "从 " << start << " 到 " << target << " 的最短路径长度: " << dist[target] << endl;       
        // 还原路径
        vector<int> path;
        for (int v = target; v != -1; v = parent[v]) {
            path.push_back(v);
        }
        reverse(path.begin(), path.end());
    }
}
int main() {
    int n = 10;
    // 邻接表：adj[u] 存储 {v, weight}
    vector<vector<pair<int, int>>> adj(n + 1);

    // 填充数据 (根据 ASCII 图)
    adj[1].push_back({2, 10});
    adj[1].push_back({3, 1});
    adj[1].push_back({4, 5});    
    adj[3].push_back({2, 1});  // 绕路更便宜的关键！
    adj[3].push_back({6, 4});  
    adj[2].push_back({5, 2});    
    adj[6].push_back({10, 20});
    adj[6].push_back({2, 15}); // 支路    
    adj[5].push_back({7, 6});    
    adj[7].push_back({10, 2}); // 终点前最后一站
    // 运行算法：求 1 到 10 的最短路径
    dijkstra(1, 10, n, adj);

    return 0;
}

/*Floyd算法：算出图中任意两点 (i, j) 之间的最短路，可以处理负权边，只要没有负权回路。对于每一点(i,j),我们尝试每一个可能的中间点k，如果dist[i][j]>dist[i][k]+dist[k][j]，那就说明通过k绕路更便宜，于是更新账本。
                    (5)        (3)          
                [1] ----> [2] ----> [3] 
                 |                   |
                 |       (10)        | (1)
                 \-----------------> [4] */
*********************************************************************************
const int INF = 1e9; // 无穷大

void floyd(int n, vector<vector<int>>& dist) {
    // 核心：三层循环
    // 第一层 k：中间借路点（上帝视角：今天咱们试试经过节点 k 会不会更快）
    for (int k = 1; k <= n; k++) {
        // 第二、三层 i, j：图中任意两点
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                // 如果经过 k 确实更省钱，就更新
                if (dist[i][k] != INF && dist[k][j] != INF) {
                    if (dist[i][k] + dist[k][j] < dist[i][j]) {
                        dist[i][j] = dist[i][k] + dist[k][j];
                    }
                }
            }
        }
    }
}
int main() {
    int n = 4; // 假设只有 4 个点
    // 邻接矩阵初始化：自己到自己是 0，其他是 INF
    vector<vector<int>> dist(n + 1, vector<int>(n + 1, INF));
    for(int i=1; i<=n; i++) dist[i][i] = 0;
    // 添加有向边
    dist[1][2] = 5;
    dist[1][4] = 10;
    dist[2][3] = 3;
    dist[3][4] = 1;
    floyd(n, dist);
    // 输出 1 到 4 的最短距离
    // 直接走是 10，绕路 1->2->3->4 是 5+3+1 = 9
    cout << "从 1 到 4 的全图最短路是: " << dist[1][4] << endl;
    return 0;
}
```

#### ==并查集和最小生成树(如何把所有的节点用最短的线连接起来)==

```c++
/*
并查集是一种树形的数据结构，主要用于解决一些元素分组的问题，用于处理一些不相交集合的合并以及查询问题。
并查集的思想是用一个数组表示了整片森林，树的根节点唯一标识了一个结合，我们只要找到了某个元素的树根，就能确定它在哪个集合里。
🌟并查集通常只需要三个基本操作：
--init(n)：初始化，每个人最初都是自己的祖先
--find(x)：查询，顺着父节点往上爬，直到找到“祖宗”（根节点）
--unite(x, y)：合并，让一个集合的祖宗认另一个集合的祖宗当爹
“路径压缩”是并查集的灵魂优化：在找祖宗的过程中，顺便把路径上的所有人都直接挂到祖宗下面。这样下次找的时候，一步到位！
*/
**************************************************************************************
// 并查集
class UnionFind {
private:
    vector<int> parent;
public:
    // 初始化：n 个元素，初始时父节点都是自己
    UnionFind(int n) {
        parent.resize(n + 1);
        iota(parent.begin(), parent.end(), 0); // parent[i] = i
    }
    // 核心操作：查找（带路径压缩）
    int find(int x) {
        if (parent[x] == x) {
            return x;
        }
        // 路径压缩：直接把父节点设为最终的根节点
        return parent[x] = find(parent[x]); 
    }
    // 核心操作：合并
    bool unite(int x, int y) {
        int rootX = find(x);
        int rootY = find(y);
        if (rootX != rootY) {
            parent[rootX] = rootY; // rootX 认 rootY 为爹
            return true;// 合并成功
        }
        return false;// 已在同一集合，合并失败（会成环）
    }
};
struct Edge {
    int u, v, weight;
};
// 排序规则：按权值从小到大排（贪心的基础）
bool compareEdges(const Edge& a, const Edge& b) {
    return a.weight < b.weight;
}
-------------------------------------------------------------------------------------
//  Kruskal 主算法，给定 n 个点和 m 条边，求最小生成树权重
int solveKruskal(int n, vector<Edge>& edges) {
    // 第一步：按权值排序
    sort(edges.begin(), edges.end(), compareEdges);
    UnionFind uf(n);
    int mstWeight = 0;   // 累计最小生成树的权重
    int edgeCount = 0;   // 累计加入的边数
    // 第二步：遍历每一条边
    for (const auto& edge : edges) {
        // 如果这两个点不连通，就合并它们（加入生成树）
        if (uf.unite(edge.u, edge.v)) {
            mstWeight += edge.weight;
            edgeCount++;
        }
    }
    // 第三步：判断是否所有点都连通了（生成树边数应为 n-1）
    return (edgeCount == n - 1) ? mstWeight : -1; // -1 表示无法构成生成树
}
```
